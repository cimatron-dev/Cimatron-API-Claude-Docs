Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)CimApp.GetActiveDoc();
if (aDOC != null) // Check if document was found
{
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
// Create the Sketcher procedure depended on the type of the active document. For now we take part document
if (aDOC.Type == interop.CimBaseAPI.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from model container
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
//Create sketcher procedure on Modeler Model
interop.CimMdlrAPI.MdSketcher aMdSketcherProcedure =
(interop.CimMdlrAPI.MdSketcher)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmSketcherProcedure);
// Cast MdSketcher to IMdSketcher to get sketcher object
interop.CimMdlrAPI.IMdSketcher aIMdSketcher = (interop.CimMdlrAPI.IMdSketcher)aMdSketcherProcedure;
interop.CimServicesAPI.ISketcher aISketcher = (interop.CimServicesAPI.ISketcher)aIMdSketcher.Sketcher;
interop.CimServicesAPI.ISkObject aSKEllipseObj1 = aISketcher.CreateObject(interop.CimServicesAPI.SketchObjectType.cmSketchEllipse);
interop.CimServicesAPI.ISkEllipse aSkEllipse1 = (interop.CimServicesAPI.ISkEllipse)aSKEllipseObj1;
double[] aPoint = new double[2];
aPoint[0] = 0;
aPoint[1] = 0;
aSkEllipse1.Center = aPoint;
aPoint[0] = 10;
aPoint[1] = 0;
aSkEllipse1.Axis = aPoint;
aSkEllipse1.Ratio = 0.8;
aISketcher.AddObject(aSKEllipseObj1);

interop.CimServicesAPI.ISkObject aSKEllipseObj2 = aISketcher.CreateObject(interop.CimServicesAPI.SketchObjectType.cmSketchEllipse);
interop.CimServicesAPI.ISkEllipse aSkEllipse2 = (interop.CimServicesAPI.ISkEllipse)aSKEllipseObj2;
aPoint[0] = 0;
aPoint[1] = 0;
aSkEllipse2.Center = aPoint;
aPoint[0] = 10;
aPoint[1] = 0;
aSkEllipse2.Axis = aPoint;
aSkEllipse2.Ratio = 1;
aISketcher.AddObject(aSKEllipseObj2);

//Create SkConstraint type object
interop.CimServicesAPI.ISkObject aConstraintObj = aISketcher.CreateObject(interop.CimServicesAPI.SketchObjectType.cmSketchConstraint);
interop.CimServicesAPI.ISkConstraint aSkConstraint = (interop.CimServicesAPI.ISkConstraint)aConstraintObj;
//Set Constraint type to SkConstraint object
aSkConstraint.ConstraintType = interop.CimServicesAPI.SketchConstraintType.cmSketchConcentric;
aSkConstraint.SetObject(0, aSKEllipseObj1);
aSkConstraint.SetObject(1, aSKEllipseObj2);
aISketcher.AddObject(aConstraintObj);

aISketcher.Evaluate();
aMdSketcherProcedure.Execute();

//Get object corresponding to index provided.
interop.CimServicesAPI.ISkObject aObj= aSkConstraint.GetObject(1);
}
}
