Example

int ObjectToDividePID = 13;
int DividingToolPID = 472;

interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication cimApp = null;
cimApp = aAppAcc.GetApplication();

interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)cimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;


interop.CimMdlrAPI.IModel aModel = (interop.CimMdlrAPI.IModel)aMdlrModel;
interop.CimMdlrAPI.ICimEntity ObjectToDivide = aModel.GetEntityById(ObjectToDividePID, aModel);
interop.CimMdlrAPI.ICimEntity DividingTool = aModel.GetEntityById(DividingToolPID, aModel);


interop.CimMdlrAPI.IMdProcedure ProcedureDivide =
(interop.CimMdlrAPI.MdDivide)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmDivideProcedure);
interop.CimMdlrAPI.IMdDivide IMdDivide = (interop.CimMdlrAPI.IMdDivide)ProcedureDivide;

interop.CimMdlrAPI.ICimEntityList ObjectsToDivide = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
ObjectsToDivide.Add(ObjectToDivide);
interop.CimMdlrAPI.ICimEntityList DividingTools = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
DividingTools.Add(DividingTool);

IMdDivide.ObjectsToDivide = ObjectsToDivide;
IMdDivide.DividingToolsEntities = DividingTools;
IMdDivide.IsSeparate = 0;
IMdDivide.DivideFaceOption = interop.CimMdlrAPI.FacePlaneOption.AsPlane;

ProcedureDivide.Execute();
