Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)CimApp.GetActiveDoc();
if (aDOC != null) // Check if document was found
{
// get model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;
// Create the PointsOnCurve procedure depended on active part document.
if (aDOC.Type == interop.CimBaseAPI.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from model container
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
// Create the modeler procedure for the MdIntersection procedure
interop.CimMdlrAPI.IMdProcedure aProcedure =
(interop.CimMdlrAPI.IMdProcedure)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmTextCurveProcedure);

interop.CimMdlrAPI.IMdTextCurve IMdTextCurve_1 = (interop.CimMdlrAPI.IMdTextCurve)aProcedure;
IMdTextCurve_1.VerticalAlignMode = interop.CimMdlrAPI.MdTextVerticalAlignMode.cmMdTextTopVerticalAlign;
IMdTextCurve_1.MirrorMode = 0;
IMdTextCurve_1.Angle = 0;
interop.CimMdlrAPI.IMdlrModel IMdlrModel_1 = (interop.CimMdlrAPI.IMdlrModel)aActiveModel_1;
interop.CimServicesAPI.IGeomServices IGeomServices_1 = (interop.CimServicesAPI.IGeomServices)IMdlrModel_1.GetGeomServicesObj();
IMdTextCurve_1.Plane = aActiveModel_1.GetEntityById(18,aActiveModel_1);
IMdTextCurve_1.AlignMode = interop.CimMdlrAPI.MdTextAlignMode.cmMdTextLeftAlign;
IMdTextCurve_1.ApproxMode = 0;
IMdTextCurve_1.BackwardsMode = 0;
IMdTextCurve_1.UpsideMode = 1;
IMdTextCurve_1.UnderlineMode = 0;
IMdTextCurve_1.VerticalMode = 1;
IMdTextCurve_1.BoldMode = 0;
IMdTextCurve_1.LineSpacing = 1;
IMdTextCurve_1.CharRatio = 1;
IMdTextCurve_1.Slant = 0;
IMdTextCurve_1.CharSpacing = 1;
IMdTextCurve_1.FontSize = 10;
IMdTextCurve_1.Text = "AKSHAY";
IMdTextCurve_1.FontName = "Arial Black";
IMdTextCurve_1.SingleLineOption = false;
IMdTextCurve_1.SingleLineOption = false;
IMdTextCurve_1.TextHeightOption = true;
IMdTextCurve_1.TextAutoScaleOption = false;
IMdTextCurve_1.SmoothAngle = 0;
IMdTextCurve_1.TextFlipOption = false;
IMdTextCurve_1.TextNormalToFaceOption = false;
IMdTextCurve_1.TextNormalToPlaneOption = false;
IMdTextCurve_1.TextMergeOption = true;
aProcedure.Execute();
}
}
