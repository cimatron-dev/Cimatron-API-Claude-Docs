interop.CimAppAccess.AppAccess AppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)AppAcc.GetApplication();
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aContainer = (interop.CimMdlrAPI.IModelContainer)aDoc;
interop.CimMdlrAPI.IAssemblyModel aAssemblyModel = (interop.CimMdlrAPI.IAssemblyModel)aContainer.Model;
//Get Instance and Model List from the Assembly model
List<interop.CimMdlrAPI.IModel> aModelList = new List<interop.CimMdlrAPI.IModel>();
List<interop.CimMdlrAPI.IAssInstance> aAssInstList = new List<interop.CimMdlrAPI.IAssInstance>();
object[] aInstlList = (object[])aAssemblyModel.GetInstances();
int c = aInstlList.Length;
foreach(object aInst in aInstlList)
{
interop.CimMdlrAPI.IAssInstance aInstance = aInst as interop.CimMdlrAPI.IAssInstance;
if (aInstance != null)
{
interop.CimMdlrAPI.IModel aTempModel = aInstance.Model;
aAssInstList.Add(aInstance);
aModelList.Add(aTempModel);
}
}
