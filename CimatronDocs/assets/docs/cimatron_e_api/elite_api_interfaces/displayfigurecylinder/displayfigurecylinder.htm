Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
interop.CimBaseAPI.ICimDocument aDoc = (interop.CimBaseAPI.ICimDocument)aCimApp.GetActiveDoc();
interop.CimServicesAPI.IInteractionSink aInteractionSink = (interop.CimServicesAPI.IInteractionSink)aDoc;
interop.CimServicesAPI.IInteraction aDisplayFigureFactoryInteraction = aInteractionSink.CreateInteraction(interop.CimServicesAPI.InteractionType.cmDisplayFigureFactory);
interop.CimServicesAPI.DisplayFigureFactory aDisplayFigureFactory = (interop.CimServicesAPI.DisplayFigureFactory)aDisplayFigureFactoryInteraction;
interop.CimServicesAPI.DisplayFigure aDisplayFigure =aDisplayFigureFactory.NewFigure();
interop.CimServicesAPI.IDisplayFigureAttribute aDisplayFigureAttribute =aDisplayFigure.CreatePrimitive(interop.CimServicesAPI.PrimitiveType.cmDisplayFigureCylinder);
interop.CimServicesAPI.DisplayFigureCylinder aCylinder = (interop.CimServicesAPI.DisplayFigureCylinder)aDisplayFigureAttribute;
aCylinder.Height = 500;
aCylinder.Radius = 100;
aDisplayFigureAttribute.Color = 80;
aDisplayFigure.Show = 1;
aDisplayFigure.Transparency = 1;
aDisplayFigure.RenderMode = interop.CimServicesAPI.RenderOption.cmRenderTransparency;
aDisplayFigureFactory.PushFigure(aDisplayFigure);
aDisplayFigureFactory.Refresh();
