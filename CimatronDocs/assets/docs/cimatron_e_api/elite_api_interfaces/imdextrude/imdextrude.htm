Example 1 Extrude with Extrude on Single SideExtrude with Extrude on Single Sideint SketchPID = 35;

interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication cimApp = null;
cimApp = aAppAcc.GetApplication() as interop.CimatronE.IApplication;

interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)cimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;


interop.CimMdlrAPI.IModel aModel = (interop.CimMdlrAPI.IModel)aMdlrModel;
interop.CimMdlrAPI.ICimEntity aSketch = aModel.GetEntityById(SketchPID, aModel);


interop.CimMdlrAPI.IMdProcedure ProcedureExt =
(interop.CimMdlrAPI.MdExtrude)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmExtrudeProcedure);
interop.CimMdlrAPI.IMdExtrude IMdExtrude = (interop.CimMdlrAPI.IMdExtrude)ProcedureExt;

IMdExtrude.Contour = aSketch;
IMdExtrude.Delta = 50;
IMdExtrude.Mode = interop.CimMdlrAPI.ExtrudeSweepMode.cmExtrudeSweepModeNew;
IMdExtrude.SideOption = interop.CimMdlrAPI.ExtrudeSideOption.cmExtrudeOneSide;

IMdExtrude.DraftAngle = 0;
IMdExtrude.DraftSideOption = interop.CimMdlrAPI.ExtrudeDraftSideOption.cmExtrudeDraftOutside;
IMdExtrude.InvertOption = interop.CimMdlrAPI.ExtrudeInvertOption.cmExtrudeForward;

ProcedureExt.Execute();
