IMdStock

This interface obtains the overall size of a part or create a bounding box (stock) around it. In 2D optimization mode, the box is based along the directions of the active UCS. In 3D optimization mode, the system analyzes and calculates the box orientation and the resulting stock is the minimum bounding box or bounding cylinder around the part. The stock size can be manually manipulated. Using this function provides an easy way for a designer to find a molding orientation and for a CNC operator to set up a direction on a mill.

// create Cimatron App Access object

//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

//Create Stock type procedure
interop.CimMdlrAPI.MdStock aMdStockProcedure =
(interop.CimMdlrAPI.MdStock)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmStockProcedure);

interop.CimMdlrAPI.IMdStock aIMdStock = (interop.CimMdlrAPI.IMdStock)aMdStockProcedure;

//Get Entity by Pids
interop.CimMdlrAPI.ICimEntity aFirstEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(13, aIModel);

interop.CimMdlrAPI.ICimEntityList aEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();

aEntities.Add(aFirstEntity);

//Get the Active UCS by Pid
interop.CimBaseAPI.ICimEntity aEntityUCS = (interop.CimBaseAPI.ICimEntity)aIModel.GetEntityById(2, aIModel);
interop.CimBaseAPI.IUcs aUcs = aEntityUCS as interop.CimBaseAPI.IUcs;


//Set/assign required properties and methods
aIMdStock.Entities = aEntities;
aIMdStock.StockOptimization = interop.CimMdlrAPI.StockOptimization.cm2DOptimization;
aIMdStock.StockResultType = interop.CimMdlrAPI.StockResultType.cmCreateSolid;
aIMdStock.StockUnits = interop.CimMdlrAPI.StockUnits.cmStockmm;
aIMdStock.StockActiveUCS = (interop.CimMdlrAPI.IUcs)aUcs;
aIMdStock.StockShapeType = interop.CimMdlrAPI.StockShapeType.cmStockCylinder;
aIMdStock.ZAxisAngle = 0;
aIMdStock.StockUCS = interop.CimMdlrAPI.StockUCS.cmStockCreateUCS;
aIMdStock.StockOffsetType = interop.CimMdlrAPI.StockOffsetType.cmStockSymmetric;
aIMdStock.StockCalculateCylinderAndBox = interop.CimMdlrAPI.StockCalculateCylinderAndBox.cmStockCalculateCylinder;
aIMdStock.StockRoundingOnOff = interop.CimMdlrAPI.StockRoundingOnOff.cmStockRoundingOff;
aIMdStock.StockFinalSize = interop.CimMdlrAPI.StockFinalSize.cmStockAuto;
aIMdStock.StockSaveSizeOnOff = interop.CimMdlrAPI.StockSaveSizeOnOff.cmStockSaveOff;
aIMdStock.StockCuttingObject = interop.CimMdlrAPI.StockCuttingObject.cmStockMarkCuttingObject;
aMdStockProcedure.Execute();
}
}
}
