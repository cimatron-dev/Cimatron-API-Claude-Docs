}

public void OnToolPushed(object iToolHelper)
{
if (iToolHelper != null)
{
m_PickToolHelper = (interop.CimServicesAPI.IPickTool)iToolHelper;
}

//Set selection filter and enable point selection

{
m_PickToolHelper.SetFilter(m_EnttFilter, 1);
}

if (m_PointFilter != null)
{
m_PickToolHelper.SetFilter((interop.CimServicesAPI.IEntityFilter)m_PointFilter, 1);
}

m_EnttLst = null;
}
