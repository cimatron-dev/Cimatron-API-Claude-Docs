using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

using interop.CimBaseAPI;

namespace PluginCommand_Training
{
public class NewPluginCommand : CimUIInfrastructure.PlugIn.ICimApiCommandPlugin
{
public CimUIInfrastructure.PlugIn.ApiCommand AppendCommand()
{
CimUIInfrastructure.PlugIn.ApiCommand Command1 = new
CimUIInfrastructure.PlugIn.ApiCommand()
{
// Set Command name
Name = "A UCS",
// Set Toolbar name
ToolbarName = "A UCS",
// Set Main-Menu/Group name
MenuPath = "MainMenu\nGroup1",
// Set Caption string
Caption = "A UCS",
// Set Tooltip string
ToolTip = "Creates A UCS",
// Set Description string
Description = "Make Copy of Main UCS at 10mm in Z direction",
// Set visibility file type
Application = CimUIInfrastructure.PlugIn.ApiApplications.Part,
// Call the execute class
ExecuteCommand = new MyExecuteCommand()
};
// Return command object
return Command1;
}
}
public class MyExecuteCommand : CimUIInfrastructure.Commands.ICimWpfCommand
{
public string GetAccelerator()
{
throw new NotImplementedException();
}
public bool OnCommand()
{
// Get Instance of CimatronE Application as mentioned in Step 5
interop.CimServicesAPI.CimApplicationProvider appProvider = new
interop.CimServicesAPI.CimApplicationProvider();
interop.CimatronE.IApplication cimApp = null;
cimApp = (interop.CimatronE.IApplication)appProvider.GetApplication();
if (cimApp == null)
{
MessageBox.Show("Application was not accessed");
return false;
}
///Code to Generate Desired UCS.
interop.CimBaseAPI.ICimDocument aDOC =
(interop.CimBaseAPI.ICimDocument)cimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aModelCont =
(interop.CimMdlrAPI.IModelContainer)aDOC;
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
//Create Procedure for UCS
interop.CimMdlrAPI.IMdProcedure ProcUCS =

(interop.CimMdlrAPI.IMdProcedure)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmUcsProcedure);
interop.CimMdlrAPI.IMdUcs imdUCs = (interop.CimMdlrAPI.IMdUcs)ProcUCS;
imdUCs.Mode = interop.CimMdlrAPI.UcsMode.cmUcsModeGeometry;
// Create 3 points for UCS
double[] point1 = new double[] { 0, 0, 10 };
double[] point2 = new double[] { 1, 0, 10 };
double[] point3 = new double[] { 0, 1, 10 };
// Assign points to UCS Procedureâs Point parameters
imdUCs.FirstPoint = point1;
imdUCs.SecondPoint = point2;
imdUCs.ThirdPoint = point3;
// Execute the process
ProcUCS.Execute();
return true;
}

CimUIInfrastructure.Commands.CimWpfUICommandStates()
{
UiState = CimUIInfrastructure.Commands.CommandUIState.Enabled
};
return states;
}
