//create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

//Create Merge type procedure
interop.CimMdlrAPI.MdMerge aMdMergeProcedure =
(interop.CimMdlrAPI.MdMerge)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmMergeProcedure);

interop.CimMdlrAPI.IMdMerge aIMdMerge = (interop.CimMdlrAPI.IMdMerge)aMdMergeProcedure;

interop.CimMdlrAPI.ICimEntity aFirstSelectedSolidEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(52, aIModel);
interop.CimMdlrAPI.ICimEntity aSecondSelectedSolidEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(17, aIModel);
interop.CimMdlrAPI.ICimEntity aThirdSelectedSolidEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(77, aIModel);
interop.CimMdlrAPI.ICimEntityList aSolidEntityList = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
aSolidEntityList.Add(aFirstSelectedSolidEntity);
aSolidEntityList.Add(aSecondSelectedSolidEntity);
aSolidEntityList.Add(aThirdSelectedSolidEntity);

//Set the required properties/methods for procedure.
aIMdMerge.Entities = aSolidEntityList;
aIMdMerge.EntDirectType[aSecondSelectedSolidEntity] = interop.CimMdlrAPI.MergeDirectType.cmMdMergeOpposite; //Second selected entity will be cleared
aIMdMerge.RemoveEntity(aThirdSelectedSolidEntity);
aMdMergeProcedure.Execute();

Example 1-Merge All Solid BodiesExample 1-Merge All Solid Bodies
import clr
#Adding references of required interop dlls
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")
clr.AddReference("interop.CimServicesAPI")
clr.AddReference("interop.CimBaseAPI")
clr.AddReference("interop.CimMdlrAPI")

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)

#Create MdDriveSolid procedure on Modeler Model
aMdMergeProcedure = (interop.CimMdlrAPI.MdMerge)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmMergeProcedure))
aIMdMerge = (interop.CimMdlrAPI.IMdMerge)(aMdMergeProcedure)

aIModel = (interop.CimMdlrAPI.IModel)(aModelCont.Model)

#Get the Solid Objects with PId
aFirstSelectedSolidEntity = aIModel.GetEntityById(52, aIModel)
aSecondSelectedSolidEntity = aIModel.GetEntityById(17, aIModel)
aThirdSelectedSolidEntity = aIModel.GetEntityById(77, aIModel)

aSolidEntityList = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
aSolidEntityList.Add(aFirstSelectedSolidEntity)
aSolidEntityList.Add(aSecondSelectedSolidEntity)
aSolidEntityList.Add(aThirdSelectedSolidEntity)

#Set / assign required properties and methods
aIMdMerge.Entities = aSolidEntityList
aIMdMerge.set_EntDirectType(aSecondSelectedSolidEntity, interop.CimMdlrAPI.MergeDirectType.cmMdMergeOpposite) #Second selected entity will be cleared
aIMdMerge.RemoveEntity(aThirdSelectedSolidEntity)

aMdMergeProcedure.Execute()
