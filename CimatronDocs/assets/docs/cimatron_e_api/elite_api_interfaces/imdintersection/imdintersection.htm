IMdIntersection

This class represents a modeler procedure for creating intersection curve of given intersecting face entities.

Example 1:Mode 1-Don't create compositeExample 1:Mode 1-Don't create composite
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
interop.CimBaseAPI.ICimDocument aDoc = (interop.CimBaseAPI.ICimDocument)CimApp.GetActiveDoc();
if (aDoc != null) // Check if document was found
{
// get model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;
// Create the PointsOnCurve procedure depended on active part document.
if (aDoc.Type == interop.CimBaseAPI.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from model container
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
// Create the modeler procedure for the MdIntersection procedure
interop.CimMdlrAPI.MdIntersection aProcedure =
(interop.CimMdlrAPI.MdIntersection)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmIntersectionProcedure);

interop.CimMdlrAPI.IMdIntersection aIMdIntersection = (interop.CimMdlrAPI.IMdIntersection)aProcedure;
interop.CimMdlrAPI.ISelection aSelection = (interop.CimMdlrAPI.ISelection)aDoc;

interop.CimMdlrAPI.ICimEntityList aFirstEntList = aSelection.Selection;
aIMdIntersection.FirstEntities = aFirstEntList;
interop.CimMdlrAPI.ICimEntityList aSecondEntList = aSelection.Selection;
aIMdIntersection.SecondEntities = aSecondEntList;
aIMdIntersection.CreateComposite = 0;
aProcedure.Execute();
}
}

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)

#Create MdPolyLine procedure on Modeler Model
aProcedure = (interop.CimMdlrAPI.MdIntersection)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmIntersectionProcedure))
aIMdIntersection = (interop.CimMdlrAPI.IMdIntersection)(aProcedure)
#//Create selection object to select entities from UI.
aSelection = (interop.CimMdlrAPI.ISelection)(aDoc)
#//Get selected entities from selection object
aFirstEntList = (interop.CimMdlrAPI.ICimEntityList)(aSelection.Selection)
aIMdIntersection.FirstEntities = aFirstEntList
//Get selected entities from selection object
aSecondEntList = (interop.CimMdlrAPI.ICimEntityList)(aSelection.Selection)
aIMdIntersection.SecondEntities = aSecondEntList
aIMdIntersection.CreateComposite = 0
aProcedure.Execute()
