IGeomServices::ApproxCurveToPrimitives

Use this property to approximate the curve into set of primitives.

Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAcc.GetApplication();
// Get active document from application
interop.CimatronE.ICimDocument aDoc = aCimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aContainer = (interop.CimMdlrAPI.IModelContainer)aDoc;
//GetModel
interop.CimMdlrAPI.IMdlrModel aMdlrModel = (interop.CimMdlrAPI.IMdlrModel)aContainer.Model;
//Get GeomService Object
interop.CimServicesAPI.IGeomServices aGeomService = (interop.CimServicesAPI.IGeomServices)aMdlrModel.GetGeomServicesObj();
interop.CimBaseAPI.IModel aModel = (interop.CimBaseAPI.IModel)aMdlrModel;
//Get Curve Using Id
interop.CimBaseAPI.ICimEntity aEnt = aModel.GetEntityById(12, aModel);
double aApproxTol = 0.01;
double aSmoothAngleTol = 0.05;
double aSharpCornerTol = 15;
//Array of IGeom3DCurve
object[] aOutList2 = aGeomService.ApproxCurveToPrimitives((interop.CimServicesAPI.ICimEntity)aEnt, aApproxTol, aSmoothAngleTol, aSharpCornerTol);

string aPath = @"C:\Temp\Points.txt";
using (StreamWriter streamWriter = new StreamWriter(aPath))
{
foreach(interop.CimServicesAPI.IGeom3DCurve aGeom in aOutList2)
{
interop.CimServicesAPI.IGeom3DCurve aCurve = aGeom as interop.CimServicesAPI.IGeom3DCurve;
if (aCurve != null)
{
double[] aStart = (double[])aCurve.Start;
double[] aEnd = (double[])aCurve.End;
double length = aCurve.Length;
streamWriter.WriteLine("Line " + $"{ aStart[0]},{aStart[1]},{aStart[2]} {aEnd[0]},{aEnd[1]},{aEnd[2]}");
}
}
}
