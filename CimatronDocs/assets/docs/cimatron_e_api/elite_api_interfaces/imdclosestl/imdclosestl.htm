IMdCloseSTL

Use this interface to close all open areas in selected mesh objects. The operation can be performed on multiple open mesh objects, including flat skins (a flat skin is a mesh where all the triangles are in the same plane).

Namespace

interop.CimMdlrAPI

Compatibility

Example 1:Close STL Auto ModeExample 1:Close STL Auto Mode // create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

//Create Drive Solid type procedure
interop.CimMdlrAPI.MdCloseSTL aMdCloseSTLProcedure =
(interop.CimMdlrAPI.MdCloseSTL)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmCloseSTLProcedure);

interop.CimMdlrAPI.IMdCloseSTL aIMdCloseSTL = (interop.CimMdlrAPI.IMdCloseSTL)aMdCloseSTLProcedure;

interop.CimMdlrAPI.ICimEntity aFirstSelectedEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(7, aIModel);
interop.CimMdlrAPI.ICimEntity aSecondSelectedEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(4, aIModel);

interop.CimMdlrAPI.ICimEntityList aSelectedEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
aSelectedEntities.Add(aFirstSelectedEntity);
aSelectedEntities.Add(aSecondSelectedEntity);

//Set / assign required properties and methods
aIMdCloseSTL.Entities = aSelectedEntities;
aIMdCloseSTL.CloseSTLMode = interop.CimMdlrAPI.CloseSTLMode.cmCloseSTLAutoMode;
aIMdCloseSTL.RemoveKeepFlatSkinMode = interop.CimMdlrAPI.RemoveKeepFlatSkinMode.cmKeepFlatSkins;
aMdCloseSTLProcedure.Execute();
}
}
}

Example 1:Close STL Auto ModeExample 1:Close STL Auto Mode import clr
#Adding references of required interop dlls
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")
clr.AddReference("interop.CimServicesAPI")
clr.AddReference("interop.CimBaseAPI")
clr.AddReference("interop.CimMdlrAPI")

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)

#Create Close Stl procedure on Modeler Model
aMdCloseSTLProcedure = (interop.CimMdlrAPI.MdCloseSTL)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmCloseSTLProcedure))
aIMdCloseSTL = (interop.CimMdlrAPI.IMdCloseSTL)(aMdCloseSTLProcedure)

aIModel = (interop.CimMdlrAPI.IModel)(aModelCont.Model)

#Get the Stl bodies with PId
aFirstSelectedEntity = aIModel.GetEntityById(7, aIModel)
aSecondSelectedEntity = aIModel.GetEntityById(4, aIModel)

aSelectedEntities = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
aSelectedEntities.Add(aFirstSelectedEntity)
aSelectedEntities.Add(aSecondSelectedEntity)

#Set / assign required properties and methods
aIMdCloseSTL.Entities = aSelectedEntities
aIMdCloseSTL.CloseSTLMode = interop.CimMdlrAPI.CloseSTLMode.cmCloseSTLAutoMode
aIMdCloseSTL.RemoveKeepFlatSkinMode = interop.CimMdlrAPI.RemoveKeepFlatSkinMode.cmKeepFlatSkins
aMdCloseSTLProcedure.Execute()
