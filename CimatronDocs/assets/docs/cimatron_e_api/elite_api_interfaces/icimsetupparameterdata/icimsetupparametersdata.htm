Remarks

The Mold and Die tabs are treated as a group.
The General tab has a number of groups.

Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)aCimApp.GetActiveDoc();
if (aDOC != null) // Check if document was found
{
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
// Create the Sketcher procedure depended on the type of the active document. For now we take part document
if (aDOC.Type == interop.CimBaseAPI.DocumentEnumType.cmPart)
{
interop.CimMdlrAPI.IModel aModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;
interop.CimServicesAPI.ICimSetupManager aSetupManagerObj = (interop.CimServicesAPI.ICimSetupManager)aModel.GetCimSetupManager();
interop.CimServicesAPI.ICimSetupData aSetupDataData = aSetupManagerObj.SetupData;
object[] aGroupListObj = (object[])aSetupDataData.ListOfGroup;
string aGroupName = (string)aGroupListObj[0];
interop.CimServicesAPI.ICimSetupGroupData aGroupData = aSetupDataData.GetGroupData(aGroupName);
object[] aGroupParamListObj = (object[])aGroupData.ListOfParameters;
string aNameOfParam = (string)aGroupParamListObj[0];
interop.CimServicesAPI.ICimSetupParametersData aParamData = aGroupData.GetCimSetupParametersData(aNameOfParam);
//=======================================================================
string aDescription = aParamData.Description;
//=======================================================================
string aGname = aParamData.GroupName;
//=======================================================================
int aParamId = aParamData.ParamId;
//=======================================================================
string aParamFormula = aParamData.GeneralParamFormula;
//=======================================================================
string aParamImage = aParamData.GeneralParamImage;
//=======================================================================
object aValue = aParamData.Value;
Type aType = aValue.GetType();
//=======================================================================
string aGeneralParamFormula = aParamData.GeneralParamFormula;
//=======================================================================
string aImagepath = aParamData.GeneralParamImage;
//=======================================================================
string aCatalogDimension = aParamData.CatalogDimension;
//=======================================================================
string aParentCatalogDim = aParamData.ParentCatalogDimension;
//=======================================================================
string aCatalogPartPath = aParamData.CatalogPartPath;
//=======================================================================
int aIsConnetedTocatalog = aParamData.IsConnectedToCatalog;
//=======================================================================
int aFormulaRoundType = aParamData.GeneralParamFormulaRoundType;
//=======================================================================
object aValueListObj = aParamData.GeneralParamValueList;
//=======================================================================
interop.CimServicesAPI.SetupValueType aValType = aParamData.GeneralParamValueType;
//=======================================================================
interop.CimServicesAPI.SetupValueUnit avalUnit = aParamData.ValueUnit;
//=======================================================================
double aFormulaIncrementalVal = aParamData.GeneralParamFormulaIncrementalVal;
//=======================================================================
}
}
