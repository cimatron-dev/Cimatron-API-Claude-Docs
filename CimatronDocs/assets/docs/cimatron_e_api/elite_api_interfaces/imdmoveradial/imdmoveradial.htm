interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
if (aCimApp != null)
{
//Get Active document from IApplication
interop.CimatronE.ICimDocument aDoc = aCimApp.GetActiveDoc();
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
//Cast ICimDocument object to ModelContainer
interop.CimMdlrAPI.IModelContainer aCont = (interop.CimMdlrAPI.IModelContainer)aDoc;
//Get model from ModelContainer and cast it to IModel.
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aCont.Model;
//Get model from ModelContainer and cast it to MdlrModel.
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aCont.Model;
//create procedure of type Radial move
interop.CimMdlrAPI.IMdProcedure aIMdProcedure = aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmMoveRadialProcedure);
interop.CimMdlrAPI.IMdMoveRadial aIMdMoveRadial = (interop.CimMdlrAPI.IMdMoveRadial)aIMdProcedure;
//Entity to be moved
interop.CimMdlrAPI.ICimEntity aCircleEntToMove = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(8, aIModel);
interop.CimMdlrAPI.ICimEntity aTriangleEntToMove = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(34, aIModel);
interop.CimBaseAPI.ICimEntityList aEntList = new interop.CimBaseAPI.CimEntityList();
aEntList.Add((interop.CimBaseAPI.ICimEntity)aCircleEntToMove);
aEntList.Add((interop.CimBaseAPI.ICimEntity)aTriangleEntToMove);
//Axis entity about which radial movement to be done
interop.CimMdlrAPI.ICimEntity aAxisEnt = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(20, aIModel);
aIMdMoveRadial.AngleValue = 90;
aIMdMoveRadial.Axis_v2 = aAxisEnt;
aIMdMoveRadial.DirectionSide = 1;
aIMdMoveRadial.Entities = (interop.CimMdlrAPI.ICimEntityList)aEntList;
//aIMdMoveRadial.RemoveEntity = (interop.CimMdlrAPI.ICimEntity)aTriangleEntToMove;
aIMdProcedure.Execute();
}
}
