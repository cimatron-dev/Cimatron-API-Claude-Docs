IMdTaper

This function tapers selected faces that have a common edge with, or are intersected by, either a selected neutral contour or a neutral plane. If the contour/plane intersects the faces, the faces can be tapered on one side or both sides of the intersection

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (aCimApp != null)
{
interop.CimatronE.ICimDocument aDoc = aCimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

//Create Taper type procedure
interop.CimMdlrAPI.MdTaper aMdTaperProcedure =
(interop.CimMdlrAPI.MdTaper)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmTaperProcedure);

interop.CimMdlrAPI.IMdTaper aIMdTaper = (interop.CimMdlrAPI.IMdTaper)aMdTaperProcedure;

//Get Entity PID's
interop.CimMdlrAPI.ICimEntity aReferenceEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(49, aIModel);
interop.CimMdlrAPI.ICimEntityList aReferenceEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
aReferenceEntities.Add(aReferenceEntity);

interop.CimMdlrAPI.ICimEntity aTaperEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(26, aIModel);
interop.CimMdlrAPI.ICimEntityList aTaperEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
aTaperEntities.Add(aTaperEntity);

//Set/assign required properties and methods
aIMdTaper.ReferenceType = interop.CimMdlrAPI.TaperReferenceType.cmNaturalPlane;
aIMdTaper.ReferenceEntities = aReferenceEntities;
aIMdTaper.TaperNeutralPlaneOption = interop.CimMdlrAPI.TaperNeutralPlaneOption.cmSizeAtPlaneLevel;
aIMdTaper.TaperAutoChainSelection = interop.CimMdlrAPI.TaperAutoChainSelection.cmAutoChainSelectionOff;
aIMdTaper.TaperDirection = interop.CimMdlrAPI.TaperDirection.cmNormalToPlaneDirection;
aIMdTaper.TaperedFaces = aTaperEntities;
aIMdTaper.FlipDirection = 1;
aIMdTaper.TaperOneAndBothSideOption = interop.CimMdlrAPI.TaperOneAndBothSideOption.cmBothSide;
aIMdTaper.TaperSplitFaces = interop.CimMdlrAPI.TaperSplitFaces.cmSplitFace;
aIMdTaper.PrimaryAngle = 15;
aIMdTaper.SecondaryAngle = 10;
aIMdTaper.TaperPrimarySideOption = interop.CimMdlrAPI.TaperPrimarySideOption.cmTaperPrimaryOutside;
aIMdTaper.TaperSecondarySideOption = interop.CimMdlrAPI.TaperSecondarySideOption.cmTaperSecondaryInside;
aMdTaperProcedure.Execute();
}
}
