Example 4: Export Selected Entities on Cimatron UI to .igs formatExport Selected Entities on Cimatron UI to .igs format
//Prerequesite: Select Entities on CimatronE UI which you want to export
interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAcc.GetApplication();
//Get Active Cimatron Document
interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)aCimApp.GetActiveDoc();
//Get Model Container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
//Get Model
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.ICimEntityList aEntityList = null;
//Get Selected Entities
interop.CimMdlrAPI.ISelection aSelection = (interop.CimMdlrAPI.ISelection)aDOC;
aEntityList = (interop.CimMdlrAPI.ICimEntityList)aSelection.Selection;
interop.CIMMiscAPI.IDI aDI = aCimApp.GetDI();
aDI.ExportEntities((interop.CIMMiscAPI.ICimEntityList)aEntityList /*Entity List*/,
@"C:\CimProject\DITest\ExportTest" /*Path to export*/,
interop.CIMMiscAPI.EDIApplicationType.cmIGES /*Export application type*/);

# Exporting Selected Entities on Cimatron UI to IGES format
# Prerequesite: Select Entities on CimatronE UI which you want to export

#import clr module
import clr
#add reference to required interop dll
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")
clr.AddReference("interop.CimMdlrAPI")
clr.AddReference("interop.CIMMiscAPI")
clr.AddReference("interop.CimBaseAPI")
#import added references
import interop.CimAppAccess as CimAppAccess
import interop.CimatronE as CimatronE
import interop.CimMdlrAPI as CimMdlrAPI
import interop.CIMMiscAPI as CIMMiscAPI
import interop.CimBaseAPI as CimBaseAPI

# Get Access to Cimatron
aCimAppAccess = CimAppAccess.AppAccess()
# Get instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = aCimAppAccess.GetApplication()
aCimApp = CimatronE.IApplication(aCimApp)

aDI = CIMMiscAPI.IDI(aCimApp.GetDI()) # Get DI Object

#Get Active Doc
aDOC = CimBaseAPI.ICimDocument(aCimApp.GetActiveDoc())
#Get Model Container
aModelCont = CimMdlrAPI.IModelContainer(aDOC);
#Get Model
aMdlrModel = CimMdlrAPI.MdlrModel(aModelCont.Model);
aEntityList = None
#Get Selected Entities
aSelection = CimMdlrAPI.ISelection(aDOC)
aEntityList = CimMdlrAPI.ICimEntityList(aSelection.Selection)
aDI.ExportEntities(aEntityList,
"C:\\CimProject\\DIPython\\ExportEntitiesTest",
CIMMiscAPI.EDIApplicationType.cmIGES)
