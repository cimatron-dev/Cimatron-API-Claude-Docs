Example 1Â Simple Revolve with angle in one directionSimple Revolve with angle in one direction
int SketchPID = 11;
int AxisPID = 15;

interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication cimApp = null;
cimApp = aAppAcc.GetApplication();

interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)cimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;


interop.CimMdlrAPI.IModel aModel = (interop.CimMdlrAPI.IModel)aMdlrModel;
interop.CimMdlrAPI.ICimEntity Sketch = aModel.GetEntityById(SketchPID, aModel);
interop.CimMdlrAPI.ICimEntity Axis = aModel.GetEntityById(AxisPID, aModel);

interop.CimMdlrAPI.IMdProcedure ProcedureRevolve =
(interop.CimMdlrAPI.MdRevolve)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmRevolveProcedure);
interop.CimMdlrAPI.IMdRevolve IMdRevolve = (interop.CimMdlrAPI.IMdRevolve)ProcedureRevolve;

IMdRevolve.Contour = Sketch;
IMdRevolve.Axis = Axis;
IMdRevolve.Mode = interop.CimMdlrAPI.RevolveSweepMode.cmRevolveSweepModeNew;
IMdRevolve.ToOption = interop.CimMdlrAPI.RevolveToOption.cmRevolveOneDir;
IMdRevolve.SideOption = interop.CimMdlrAPI.RevolveSideOption.cmRevolveOneSide;
IMdRevolve.Theta = 270;

ProcedureRevolve.Execute();
