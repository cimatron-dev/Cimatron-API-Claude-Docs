IMdPointsOnSkin

Use this class to create points with a constant difference of the points, ignoring areas with a small curvature. The points are created as a point cloud.

Example 1-By Grid - Point Spread MethodExample 1-By Grid - Point Spread Method
int EntitiesPID = 7;
int aUcsId = 6;

interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication cimApp = null;
cimApp = (interop.CimatronE.IApplication)aAppAcc.GetApplication();

interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)cimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;


interop.CimMdlrAPI.IModel aModel = (interop.CimMdlrAPI.IModel)aMdlrModel;
interop.CimMdlrAPI.ICimEntity Entities = (interop.CimMdlrAPI.ICimEntity)aModel.GetEntityById(EntitiesPID, aModel);

interop.CimMdlrAPI.ICimEntityList EntitiesForPointCreation = (interop.CimMdlrAPI.ICimEntityList) new interop.CimBaseAPI.CimEntityList();

EntitiesForPointCreation.Add(Entities);


interop.CimMdlrAPI.MdPointsOnSkin aPointsOnSkinProcedure =
(interop.CimMdlrAPI.MdPointsOnSkin)
aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmPointsOnSkinProcedure);
interop.CimMdlrAPI.IMdPointsOnSkin IMdPointsOnSkin = (interop.CimMdlrAPI.IMdPointsOnSkin)aPointsOnSkinProcedure;


interop.CimBaseAPI.ICimEntity aEntityUCS = (interop.CimBaseAPI.ICimEntity)aModel.GetEntityById(aUcsId, aModel);
interop.CimBaseAPI.IUcs aUcs = aEntityUCS as interop.CimBaseAPI.IUcs;

IMdPointsOnSkin.Entities = EntitiesForPointCreation;
IMdPointsOnSkin.DistanceBetweenPoints = 2.0;
IMdPointsOnSkin.ReferenceUCS = (interop.CimMdlrAPI.IUcs)aUcs;
IMdPointsOnSkin.IncludeVerticalWalls = 1;
IMdPointsOnSkin.MinimumCurvature = 5.5;
IMdPointsOnSkin.PointsSpreadMethod = interop.CimMdlrAPI.MdPointsOnSkinPointsSpread.cmPointsOnSkinByGrid;
aPointsOnSkinProcedure.Execute();

Example 1-By Grid - Point Spread MethodExample 1-By Grid - Point Spread Method
import clr
#Adding references of required interop dlls
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")
clr.AddReference("interop.CimServicesAPI")
clr.AddReference("interop.CimBaseAPI")
clr.AddReference("interop.CimMdlrAPI")

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

EntitiesPID = 7
ReferenceUCSPID = 2

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)

#Create MdPointsOnSkin procedure on Modeler Model
aPointsOnSkinProcedure = (interop.CimMdlrAPI.MdPointsOnSkin)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmPointsOnSkinProcedure))
IMdPointsOnSkin = (interop.CimMdlrAPI.IMdPointsOnSkin)(aPointsOnSkinProcedure)

aIModel = (interop.CimMdlrAPI.IModel)(aModelCont.Model)
#Get the face/body/mesh entity with PId .
Entities = aIModel.GetEntityById(EntitiesPID, aIModel)

EntitiesForPointCreation = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
EntitiesForPointCreation.Add(Entities)

aReferenceUCS = interop.CimMdlrAPI.IUcs(aIModel.GetEntityById(ReferenceUCSPID, aIModel))

#Set Properties
IMdPointsOnSkin.Entities = EntitiesForPointCreation
IMdPointsOnSkin.DistanceBetweenPoints = 2.0
IMdPointsOnSkin.ReferenceUCS = aReferenceUCS
IMdPointsOnSkin.IncludeVerticalWalls = 0
IMdPointsOnSkin.MinimumCurvature = 5.5
IMdPointsOnSkin.PointsSpreadMethod = interop.CimMdlrAPI.MdPointsOnSkinPointsSpread.cmPointsOnSkinByTriangulation
aPointsOnSkinProcedure.Execute()
