IMdPoint

Use this interface to create points by their coordinates. It is possible to create more than one point in each call of a procedure.

Namespace

Compatibility

Cimatron 2024 or later

Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
if (CimApp != null)
{
// Get active document from application
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Get model container from document
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the MdPoint procedure depended on the type of the active document.
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

interop.CimMdlrAPI.MdPoint aMdPointProcedure =
(interop.CimMdlrAPI.MdPoint)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmPointProcedure);

interop.CimMdlrAPI.IMdPoint aIMdPoint = (interop.CimMdlrAPI.IMdPoint)aMdPointProcedure;
double[] aPointsArr = new double[6];
aPointsArr[0] = 20;
aPointsArr[1] = 20;
aPointsArr[2] = 0;
aPointsArr[3] = 20;
aPointsArr[4] = 20;
aPointsArr[5] = 0;
aIMdPoint.Points = aPointsArr;

aMdPointProcedure.Execute();

//Get Points data from proc object
object aObj = aIMdPoint.Points;
object[] aObjArr = (object[])aObj;
foreach(var item in aObjArr)
{
interop.CimMdlrAPI.IPointData aPoint = (interop.CimMdlrAPI.IPointData)item;
}
}
}
}
