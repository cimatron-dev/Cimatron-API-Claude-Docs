IMdDivideByContour

Use this interface to divide solid object by sweeping a contour. This interface divides one or more objects by extruding a closed sketch or composite curve on one or both sides, by a given value and direction.

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

//Create Divide By Contour type procedure
interop.CimMdlrAPI.MdDivideByContour aMdDivideByContourProcedure =
(interop.CimMdlrAPI.MdDivideByContour)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmDivideByContourProcedure);

interop.CimMdlrAPI.IMdDivideByContour aIMdDivideByContour = (interop.CimMdlrAPI.IMdDivideByContour)aMdDivideByContourProcedure;
interop.CimMdlrAPI.ICimEntity aFirstEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(13, aIModel);
interop.CimMdlrAPI.ICimEntity aFace = aIModel.GetEntityById(82, aIModel);

interop.CimMdlrAPI.ICimEntityList aObjectEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();

aObjectEntities.Add(aFirstEntity);

//Set/assign required properties and methods
aIMdDivideByContour.ObjectsToDivide = aObjectEntities;
aIMdDivideByContour.Contour = aFace;
aIMdDivideByContour.ToOption = interop.CimMdlrAPI.DivideToOption.cmDivideToClosest;
aIMdDivideByContour.SideOption = interop.CimMdlrAPI.DivideSideOption.cmDivideOneSide;
aIMdDivideByContour.DraftAngle = 2.0;
aIMdDivideByContour.DraftSideOption = interop.CimMdlrAPI.DivideDraftSideOption.cmDivideDraftInside;
double[] aDir = { 1, 0, 0 };
aIMdDivideByContour.Direction = aDir;
aMdDivideByContourProcedure.Execute();
}
}
