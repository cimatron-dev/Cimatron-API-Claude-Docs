Example

//Sample Code in CSharp Cimatron InteractionSink object for modeler.
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
//InteractionSink can be casted from interop.CimatronE.IApplication or interop.CimatronE.ICimDocument
interop.CimMdlrAPI.IInteractionSink aInteractionSink = (interop.CimMdlrAPI.IInteractionSink)aCimApp;
interop.CimMdlrAPI.IInteraction aInteraction = aInteractionSink.CreateInteraction(interop.CimMdlrAPI.InteractionType.cmUserMessage);
int aId = aInteraction.ID;
interop.CimMdlrAPI.InteractionType aType= aInteraction.Type;

import clr
#Adding references of required interop dlls
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")
clr.AddReference("interop.CimServicesAPI")
clr.AddReference("interop.CimBaseAPI")
clr.AddReference("interop.CimMdlrAPI")

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = CimatronE.IApplication(aCimAppAccess.GetApplication())
aInteractionSink = (interop.CimBaseAPI.IInteractionSink)(aCimApp)
aInteraction = (interop.CimMdlrAPI.IInteraction)(aInteractionSink.CreateInteraction(interop.CimMdlrAPI.InteractionType.cmUserMessage))
aId= aInteraction.ID
aType= (interop.CimMdlrAPI.InteractionType)(aInteraction.Type)
