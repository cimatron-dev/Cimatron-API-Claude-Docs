IMdGlobalDecimate

Use this class to decimate (simplify) the mesh objects.

Namespace

interop.CimMdlrAPI

Compatibility

Cimatron 2025 or later

Properties

Get, Set

 MeshObjects interop.CimMdlrAPI.ICimEntityList

Get, Set

 MaximumChange double

Methods

NoneÂ 

Events

NoneÂ 

Remarks

None

Example

//-------------------------------Mesh - Global Decimate -------------------------------
int MeshObjectsPID = 4;

interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication cimApp = null;
cimApp = (interop.CimatronE.IApplication)aAppAcc.GetApplication();

interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)cimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;

interop.CimMdlrAPI.IModel aModel = (interop.CimMdlrAPI.IModel)aMdlrModel;
interop.CimMdlrAPI.ICimEntity MeshObjects = (interop.CimMdlrAPI.ICimEntity)aModel.GetEntityById(MeshObjectsPID, aModel);

interop.CimMdlrAPI.ICimEntityList MeshObj = (interop.CimMdlrAPI.ICimEntityList) new interop.CimBaseAPI.CimEntityList();

MeshObj.Add(MeshObjects);

interop.CimMdlrAPI.MdGlobalDecimate aGlobalDecimateProcedure =
(interop.CimMdlrAPI.MdGlobalDecimate)
aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmGlobalDecimateProcedure);
interop.CimMdlrAPI.IMdGlobalDecimate IMdGlobalDecimate = (interop.CimMdlrAPI.IMdGlobalDecimate)aGlobalDecimateProcedure;

IMdGlobalDecimate.MeshObjects = MeshObj;
IMdGlobalDecimate.MaximumChange = 0.03;
aGlobalDecimateProcedure.Execute();
