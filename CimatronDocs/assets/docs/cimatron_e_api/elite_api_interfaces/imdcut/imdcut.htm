Example

int ObjToCutPID = 94;
int CuttingObjPID = 13;

interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication cimApp = null;
cimApp = aAppAcc.GetApplication();

interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)cimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;


interop.CimMdlrAPI.IModel aModel = (interop.CimMdlrAPI.IModel)aMdlrModel;
interop.CimMdlrAPI.ICimEntity ObjToCut = aModel.GetEntityById(ObjToCutPID, aModel);
interop.CimMdlrAPI.ICimEntity CuttingObj = aModel.GetEntityById(CuttingObjPID, aModel);

interop.CimMdlrAPI.ICimEntityList ObjsToCut = (interop.CimMdlrAPI.ICimEntityList) new interop.CimBaseAPI.CimEntityList();
interop.CimMdlrAPI.ICimEntityList CuttingObjs = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();

ObjsToCut.Add(ObjToCut);
CuttingObjs.Add(CuttingObj);


interop.CimMdlrAPI.MdCut aCutProcedure =
(interop.CimMdlrAPI.MdCut)
aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmCutProcedure);
interop.CimMdlrAPI.IMdCut IMdCut = (interop.CimMdlrAPI.IMdCut)aCutProcedure;

IMdCut.ObjectsToCut = ObjsToCut;
IMdCut.CuttingToolsEntities = CuttingObjs;
IMdCut.DirectionSide = 0; // Set 0 or 1 only if cutting surface is plane/face
aCutProcedure.Execute();

ObjsToCut = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
