Example 1-Cut/Divide with Cut on BlankExample 1-Cut/Divide with Cut on Blank
int MeshObjectsToCutPID = 7;
int CuttingGeometryPID = 44;

interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication cimApp = null;
cimApp = (interop.CimatronE.IApplication)aAppAcc.GetApplication();

interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)cimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;


interop.CimMdlrAPI.IModel aModel = (interop.CimMdlrAPI.IModel)aMdlrModel;
interop.CimMdlrAPI.ICimEntity MeshObjectsToCut = (interop.CimMdlrAPI.ICimEntity)aModel.GetEntityById(MeshObjectsToCutPID, aModel);
interop.CimMdlrAPI.ICimEntity CuttingGeometry = aModel.GetEntityById(CuttingGeometryPID, aModel);

interop.CimMdlrAPI.ICimEntityList MeshObjToCut = (interop.CimMdlrAPI.ICimEntityList) new interop.CimBaseAPI.CimEntityList();

MeshObjToCut.Add(MeshObjectsToCut);


interop.CimMdlrAPI.MdMeshCutOrDivide aMeshCutDivideProcedure =
(interop.CimMdlrAPI.MdMeshCutOrDivide)
aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmMeshCutOrDivideProcedure);
interop.CimMdlrAPI.IMdMeshCutOrDivide IMdMeshCutDivide = (interop.CimMdlrAPI.IMdMeshCutOrDivide)aMeshCutDivideProcedure;

IMdMeshCutDivide.MeshObjectsToCut = MeshObjToCut;
IMdMeshCutDivide.CuttingGeometry = CuttingGeometry;
IMdMeshCutDivide.ShellPreference = interop.CimMdlrAPI.MeshCutOrDivideShellPreference.cmMeshCutOrDivideDontSeperateShells;
IMdMeshCutDivide.FlipMeshNormal = 1;
IMdMeshCutDivide.OperationMethod = interop.CimMdlrAPI.MeshCutOrDivideOperationMethod.cmMeshCutOrDivideAsBoolean;
IMdMeshCutDivide.Operation = interop.CimMdlrAPI.MeshCutOrDivideOperation.cmMeshCutOrDivide_Cut;
IMdMeshCutDivide.CuttingObjectPreference = interop.CimMdlrAPI.MeshCutOrDivideCuttingObjectPreference.cmMeshCutOrDivideRemoveCuttingObject;
IMdMeshCutDivide.TransferAttributes = 1;
aMeshCutDivideProcedure.Execute();

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

MeshObjectsToCutPID = 7
CuttingGeometryPID = 44

aCimAppAccess = interop.CimAppAccess.AppAccess()
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aDOC = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDOC)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)

aModel = interop.CimMdlrAPI.IModel(aMdlrModel)
MeshObjectsToCut = aModel.GetEntityById(MeshObjectsToCutPID, aModel)
CuttingGeometry = aModel.GetEntityById(CuttingGeometryPID, aModel)

aMeshCutDivideProcedure = (interop.CimMdlrAPI.MdMeshCutOrDivide)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmMeshCutOrDivideProcedure))
IMdMeshCutDivide = (interop.CimMdlrAPI.IMdMeshCutOrDivide)(aMeshCutDivideProcedure)

MeshObjToCut = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
MeshObjToCut.Add(MeshObjectsToCut)

IMdMeshCutDivide.MeshObjectsToCut = MeshObjToCut
IMdMeshCutDivide.CuttingGeometry = CuttingGeometry
IMdMeshCutDivide.ShellPreference = interop.CimMdlrAPI.MeshCutOrDivideShellPreference.cmMeshCutOrDivideDontSeperateShells
IMdMeshCutDivide.FlipMeshNormal = 1
IMdMeshCutDivide.OperationMethod = interop.CimMdlrAPI.MeshCutOrDivideOperationMethod.cmMeshCutOrDivideAsBoolean
IMdMeshCutDivide.Operation = interop.CimMdlrAPI.MeshCutOrDivideOperation.cmMeshCutOrDivide_Cut
IMdMeshCutDivide.CuttingObjectPreference = interop.CimMdlrAPI.MeshCutOrDivideCuttingObjectPreference.cmMeshCutOrDivideRemoveCuttingObject
IMdMeshCutDivide.TransferAttributes = 1
aMeshCutDivideProcedure.Execute()
