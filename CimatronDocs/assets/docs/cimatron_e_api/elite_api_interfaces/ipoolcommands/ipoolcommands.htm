Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
interop.CimBaseAPI.IPoolCommands aPoolCommand = (interop.CimBaseAPI.IPoolCommands)aCimApp.GetPoolCommands();
string[] aNamesList = (string[]) aPoolCommand.GetCommandsNames();
string[] aPluginNamesList = (string[]) aPoolCommand.GetPluginCommandsNames();
interop.CimBaseAPI.ICimCommand aCommand= aPoolCommand.GetCommand("iCategoryName", "iCommandName");

import clr
#Adding references of required interop dlls
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")
clr.AddReference("interop.CimServicesAPI")
clr.AddReference("interop.CimBaseAPI")
clr.AddReference("interop.CimMdlrAPI")

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = CimatronE.IApplication(aCimAppAccess.GetApplication())
aPoolCommand=(interop.CimBaseAPI.IPoolCommands)(aCimApp.GetPoolCommands())
aNamesList = aPoolCommand.GetCommandsNames()
aPluginNamesList = aPoolCommand.GetPluginCommandsNames()
aCommand= (interop.CimBaseAPI.ICimCommand)(aPoolCommand.GetCommand("iCategoryName", "iCommandName"))
