Example 1: Mode 1 -By deltaExample 1: Mode 1 -By delta
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
interop.CimBaseAPI.ICimDocument aDoc = (interop.CimBaseAPI.ICimDocument)CimApp.GetActiveDoc();
if (aDoc != null) // Check if document was found
{
// get model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;
// Create the point procedure depended on active part document.
if (aDoc.Type == interop.CimBaseAPI.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

interop.CimMdlrAPI.MdCurveExtend aMdCurveExtendProcedure =
(interop.CimMdlrAPI.MdCurveExtend)
aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmCurveExtendProcedure);

interop.CimMdlrAPI.IMdCurveExtend aIMdCurveExtend = (interop.CimMdlrAPI.IMdCurveExtend)aMdCurveExtendProcedure;
interop.CimMdlrAPI.ICimEntity aCurveEnt = aIModel.GetEntityById(147, aIModel);
aIMdCurveExtend.Entity = aCurveEnt;
aIMdCurveExtend.ExtendSide = interop.CimMdlrAPI.CurveExtendSideCurveOptions.cmEndOfCurve;
aIMdCurveExtend.LinearOption = interop.CimMdlrAPI.CurveExtendLinearOptions.cmExtendCurveNatural;
aIMdCurveExtend.ReferenceOption = interop.CimMdlrAPI.CurveExtendReferenceOptions.cmExtendCurveDelta;
aIMdCurveExtend.DeltaValue = 100;
aIMdCurveExtend.NewCurveOption = 1;
aMdCurveExtendProcedure.Execute();
}
}
