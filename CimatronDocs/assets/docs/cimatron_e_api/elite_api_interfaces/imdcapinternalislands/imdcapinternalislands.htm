IMdCapInternalIslands

This Interface will close open gaps in a solid object, or create internal parting faces in the parting surface part by closing gaps on the work part faces.

Example 1-Create Faces for All Pockets And HolesExample 1-Create Faces for All Pockets And Holes
// PREREQUISITE:
// 1. Open Cimatron and Create a Model.
// 2. In created Model place some Through Holes, Blind Holes and Pockets.
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

//Create Cap Internal Islands type procedure
interop.CimMdlrAPI.MdCapInternalIslands aMdCapInternalIslandsProcedure =
(interop.CimMdlrAPI.MdCapInternalIslands)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmCapInternalIslandsProcedure);

interop.CimMdlrAPI.IMdCapInternalIslands aIMdCapInternalIslands = (interop.CimMdlrAPI.IMdCapInternalIslands)aMdCapInternalIslandsProcedure;

//Get Entity by Pids
interop.CimMdlrAPI.ICimEntity aFirstSelectedEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(90, aIModel);
interop.CimMdlrAPI.ICimEntity aSecondSelectedEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(18, aIModel);

//Create Entity List and Add each entities to the List
interop.CimMdlrAPI.ICimEntityList aSelectedEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
aSelectedEntities.Add(aFirstSelectedEntity);
aSelectedEntities.Add(aSecondSelectedEntity);

//Set/assign required properties and methods
aIMdCapInternalIslands.Entities = aSelectedEntities;
aIMdCapInternalIslands.CapInternalIslandsSingleMultiObjects = interop.CimMdlrAPI.CapInternalIslandsSingleMultiObjects.cmMultiObjects;
aIMdCapInternalIslands.CapInternalIslandsCreateAndDonotCreateWires = interop.CimMdlrAPI.CapInternalIslandsCreateAndDonotCreateWires.cmDoNotCreateWires;
aIMdCapInternalIslands.CapInternalIslandsTypeOfHole = interop.CimMdlrAPI.CapInternalIslandsTypeOfHole.cmAllPocketsandHoles;
aMdCapInternalIslandsProcedure.Execute();
}

}
}

# PREREQUISITE:
# 1. Open Cimatron and Create a Model.

import clr
#Adding references of required interop dlls
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)

#Create Cap Internal Islands procedure on Modeler Model
aMdCapInternalIslandsProcedure = (interop.CimMdlrAPI.MdCapInternalIslands)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmCapInternalIslandsProcedure))
aIMdCapInternalIslands = (interop.CimMdlrAPI.IMdCapInternalIslands)(aMdCapInternalIslandsProcedure)

aIModel = (interop.CimMdlrAPI.IModel)(aModelCont.Model)

#Get Entity by Pids
aFirstEntity = aIModel.GetEntityById(90, aIModel)
aSecondEntity = aIModel.GetEntityById(18, aIModel)

#Create Entity List and Add each entities to the List
aSelectedEntities = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
aSelectedEntities.Add(aFirstEntity)
aSelectedEntities.Add(aSecondEntity)

#Set / assign required properties and methods
aIMdCapInternalIslands.Entities = aSelectedEntities
aIMdCapInternalIslands.CapInternalIslandsSingleMultiObjects = interop.CimMdlrAPI.CapInternalIslandsSingleMultiObjects.cmMultiObjects
aIMdCapInternalIslands.CapInternalIslandsCreateAndDonotCreateWires = interop.CimMdlrAPI.CapInternalIslandsCreateAndDonotCreateWires.cmDoNotCreateWires
aIMdCapInternalIslands.CapInternalIslandsTypeOfHole = interop.CimMdlrAPI.CapInternalIslandsTypeOfHole.cmAllPocketsandHoles
aMdCapInternalIslandsProcedure.Execute()

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)

#Create Cap Internal Islands procedure on Modeler Model
aMdCapInternalIslandsProcedure = (interop.CimMdlrAPI.MdCapInternalIslands)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmCapInternalIslandsProcedure))
aIMdCapInternalIslands = (interop.CimMdlrAPI.IMdCapInternalIslands)(aMdCapInternalIslandsProcedure)

aIModel = (interop.CimMdlrAPI.IModel)(aModelCont.Model)

#Get Entity by Pids
aFirstEntity = aIModel.GetEntityById(90, aIModel)
aSecondEntity = aIModel.GetEntityById(18, aIModel)
aThirdEntity = aIModel.GetEntityById(204, aIModel)
aFourthEntity = aIModel.GetEntityById(207, aIModel)
aFifthEntity = aIModel.GetEntityById(238, aIModel)

#Create Entity List and Add each entities to the List
aSelectedEntities = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
aSelectedEntities.Add(aFirstEntity)
aSelectedEntities.Add(aSecondEntity)
aSelectedEntities.Add(aThirdEntity)
aSelectedEntities.Add(aFourthEntity)
aSelectedEntities.Add(aFifthEntity)

#Set / assign required properties and methods
aIMdCapInternalIslands.Entities = aSelectedEntities
aIMdCapInternalIslands.CapInternalIslandsSingleMultiObjects = interop.CimMdlrAPI.CapInternalIslandsSingleMultiObjects.cmMultiObjects
aIMdCapInternalIslands.CapInternalIslandsCreateAndDonotCreateWires = interop.CimMdlrAPI.CapInternalIslandsCreateAndDonotCreateWires.cmCreateWiresOnLoopsthatCannotBeClosed
aIMdCapInternalIslands.CapInternalIslandsTypeOfHole = interop.CimMdlrAPI.CapInternalIslandsTypeOfHole.cmAllPocketsandHoles
aMdCapInternalIslandsProcedure.Execute()
