IMdDriveSolid

Drive a contour to add or remove material. Use the Drive(Solid) function to:

  1. Create a new solid object

  2. Add to an existing solid object

  3. Remove material from a solid object by driving a 2D sketch along a spine

Example 1-Drive Solid for Normal OptionExample 1-Drive Solid for Normal Option
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

//Create Drive Solid type procedure
interop.CimMdlrAPI.MdDriveSolid aMdDriveSolidProcedure =
(interop.CimMdlrAPI.MdDriveSolid)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmDriveSolidProcedure);

interop.CimMdlrAPI.IMdDriveSolid aIMdDriveSolid = (interop.CimMdlrAPI.IMdDriveSolid)aMdDriveSolidProcedure;
interop.CimMdlrAPI.ICimEntity aContourSection = aIModel.GetEntityById(16, aIModel);
interop.CimMdlrAPI.ICimEntity aContourSpine = aIModel.GetEntityById(27, aIModel);

//Set/assign required properties and methods
aIMdDriveSolid.Section = aContourSection;
aIMdDriveSolid.Spine = aContourSpine;
aIMdDriveSolid.ParallelNormal = interop.CimMdlrAPI.DriveSolidMode.cmDriveSolidNormal;
aIMdDriveSolid.Mode = interop.CimMdlrAPI.DriveSolidSweepMode.cmDriveSolidSweepModeNew;
aIMdDriveSolid.Invert = interop.CimMdlrAPI.DriveSolidInvert.cmDriveSolidForward;
aIMdDriveSolid.ThinWall = 1;
aIMdDriveSolid.ThinWallMode = interop.CimMdlrAPI.DriveSolidThinWallMode.cmDriveSolidOneDir;
aIMdDriveSolid.Thickness = 2;
aIMdDriveSolid.CornerOption = interop.CimMdlrAPI.DriveSolidCornerOption.cmDriveSolidRoundCorner;
aIMdDriveSolid.CornerRadius = 2;
aIMdDriveSolid.ConstraintPlaneMode = interop.CimMdlrAPI.DriveSolidConstraintPlaneMode.cmDriveSolidPlaneZReferenced;
aIMdDriveSolid.DraftSideOption = interop.CimMdlrAPI.DriveSolidDraftSideOption.cmDriveSolidDraftOutside;
aIMdDriveSolid.DraftAngle = 0;

aMdDriveSolidProcedure.Execute();
}
}
}
