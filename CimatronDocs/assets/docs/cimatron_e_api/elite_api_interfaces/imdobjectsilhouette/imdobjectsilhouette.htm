//Sample Code in CSharp to get MdObjectSilhouette object.
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)aCimApp.GetActiveDoc();
if (aDOC != null) // Check if document was found
{
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
// Create the Sketcher procedure depended on the type of the active document. For now we take part document
if (aDOC.Type == interop.CimBaseAPI.DocumentEnumType.cmPart)
{
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;
// Create the modeler procedure for the Object Silhouette procedure
interop.CimMdlrAPI.IMdProcedure aObjectSilhouetteProcedure =
(interop.CimMdlrAPI.MdObjectSilhouette)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmObjectSilhouetteProcedure);

interop.CimMdlrAPI.ICimEntity aPlane = aIModel.GetEntityById(18, aIModel);

interop.CimMdlrAPI.IMdObjectSilhouette aIMdObjectSilhouette = (interop.CimMdlrAPI.IMdObjectSilhouette)aObjectSilhouetteProcedure;

//Select body type entity to project
interop.CimBaseAPI.ICimEntity aObject = (interop.CimBaseAPI.ICimEntity)aIModel.GetEntityById(10, aIModel);
interop.CimBaseAPI.ICimEntityList aObjectList = new interop.CimBaseAPI.CimEntityList();
aObjectList.Add(aObject);
aIMdObjectSilhouette.ObjectEntities = (interop.CimMdlrAPI.ICimEntityList)aObjectList;
aIMdObjectSilhouette.Offset = 25;
aIMdObjectSilhouette.Tolerance = 0.02;
//Select the type as per object type
aIMdObjectSilhouette.ShapeType = interop.CimMdlrAPI.SilhouetteShapeType.cmProjectOutline;
aIMdObjectSilhouette.ObjectPortion = interop.CimMdlrAPI.SilhouetteMainOption.cmSilhouetteGlobal;
aIMdObjectSilhouette.OutputType = interop.CimMdlrAPI.SilhouetteOutputType.cmOuterAndInner;
aIMdObjectSilhouette.ProjectionEntity = aPlane;
aIMdObjectSilhouette.SilhouettePlaneDirectionOption = interop.CimMdlrAPI.SilhouetteDirection.cmSilhouetteDir;
aObjectSilhouetteProcedure.Execute();
}
}
