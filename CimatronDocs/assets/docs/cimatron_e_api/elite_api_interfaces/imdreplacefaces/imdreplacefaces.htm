IMdReplaceFaces

Use this interface to replace a face (or set of faces) with another face (or set of faces). The Replace option comes under Extend Object Command in Solid Procedures.

Namespace

interop.CimMdlrAPI

Compatibility

Cimatron 2024 or later

Example

Example 1-Replace Faces Include Smooth FacesExample 1-Replace Faces Include Smooth Faces
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;
// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;
//Create Replace Faces type procedure
interop.CimMdlrAPI.MdReplaceFaces aMdReplaceFacesProcedure =
(interop.CimMdlrAPI.MdReplaceFaces)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmReplaceFacesProcedure);
interop.CimMdlrAPI.IMdReplaceFaces aIMdReplaceFaces = (interop.CimMdlrAPI.IMdReplaceFaces)aMdReplaceFacesProcedure;
//Get Entity by Pids
interop.CimMdlrAPI.ICimEntity aFirstSourceEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(235, aIModel);
interop.CimMdlrAPI.ICimEntity aSecondSourceEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(182, aIModel);
interop.CimMdlrAPI.ICimEntity aFirstTargetEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(85, aIModel);
interop.CimMdlrAPI.ICimEntity aSecondTargetEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(82, aIModel);
interop.CimMdlrAPI.ICimEntityList aSourceEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
interop.CimMdlrAPI.ICimEntityList aTargetEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
//Add Source Entities
aSourceEntities.Add(aFirstSourceEntity);
aSourceEntities.Add(aSecondSourceEntity);
//Add Target Entities
aTargetEntities.Add(aFirstTargetEntity);
aTargetEntities.Add(aSecondTargetEntity);
//Set/assign required properties and methods
aIMdReplaceFaces.ReplaceOption = interop.CimMdlrAPI.ReplaceOption.cmReplaceOption;
aIMdReplaceFaces.SourceEntities = aSourceEntities;
aIMdReplaceFaces.TargetEntities = aTargetEntities;
aIMdReplaceFaces.KeepRemoveTargetFaces = interop.CimMdlrAPI.KeepRemoveTargetFaces.cmKeepTargetFaces;
aIMdReplaceFaces.SmoothFaces = interop.CimMdlrAPI.SmoothFaces.cmIncludeSmoothFaces;
aIMdReplaceFaces.OffsetDistance = 3;
aIMdReplaceFaces.MergeAdjacentEntities = interop.CimMdlrAPI.MergeAdjacentEntities.cmMergeAdjacentFaces_On;
aMdReplaceFacesProcedure.Execute();
}
}
}

import clr
#Adding references of required interop dlls
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")
clr.AddReference("interop.CimServicesAPI")
clr.AddReference("interop.CimBaseAPI")
clr.AddReference("interop.CimMdlrAPI")
#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI
#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()
#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())
aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)
#Create MdRemoveFaces procedure on Modeler Model
aMdReplaceFacesProcedure = (interop.CimMdlrAPI.MdReplaceFaces)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmReplaceFacesProcedure))
aIMdReplaceFaces = (interop.CimMdlrAPI.IMdReplaceFaces)(aMdReplaceFacesProcedure)
aIModel = (interop.CimMdlrAPI.IModel)(aModelCont.Model)
#Get Entity by Pids
aFirstSourceEntity = aIModel.GetEntityById(235, aIModel)
aSecondSourceEntity = aIModel.GetEntityById(182, aIModel)
aFirstTargetEntity = aIModel.GetEntityById(85, aIModel)
aSecondTargetEntity = aIModel.GetEntityById(82, aIModel)
aSourceEntities = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
aTargetEntities = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
#Add Source Entities
aSourceEntities.Add(aFirstSourceEntity)
aSourceEntities.Add(aSecondSourceEntity)
#Add Target Entities
aTargetEntities.Add(aFirstTargetEntity)
aTargetEntities.Add(aSecondTargetEntity)
#Set / assign required properties and methods
aIMdReplaceFaces.ReplaceOption = interop.CimMdlrAPI.ReplaceOption.cmReplaceOption
aIMdReplaceFaces.SourceEntities = aSourceEntities
aIMdReplaceFaces.TargetEntities = aTargetEntities
aIMdReplaceFaces.KeepRemoveTargetFaces = interop.CimMdlrAPI.KeepRemoveTargetFaces.cmKeepTargetFaces
aIMdReplaceFaces.SmoothFaces = interop.CimMdlrAPI.SmoothFaces.cmIncludeSmoothFaces
aIMdReplaceFaces.OffsetDistance = 3
aIMdReplaceFaces.MergeAdjacentEntities = interop.CimMdlrAPI.MergeAdjacentEntities.cmMergeAdjacentFaces_On
aMdReplaceFacesProcedure.Execute()
