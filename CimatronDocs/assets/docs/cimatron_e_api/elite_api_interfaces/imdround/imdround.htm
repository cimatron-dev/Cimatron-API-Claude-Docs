import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aPdm = aCimApp.GetPdm()
aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())

aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)

#Create MdRound procedure on Modeler Model
aMdRoundProcedure =(interop.CimMdlrAPI.MdRound)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmRoundProcedure))
interop.CimMdlrAPI.IMdRound aIMdRound = (interop.CimMdlrAPI.IMdRound)aMdRoundProcedure
aIMdRound.ConstantType = interop.CimMdlrAPI.RoundConstantType.cmRoundConstantByChordLength
aIMdRound.RoundChord= 5
aIMdRound.Mode = interop.CimMdlrAPI.RoundMode.cmRoundConst
aSelection = (interop.CimMdlrAPI.ISelection)(aDoc)
interop.CimMdlrAPI.ICimEntityList aEntList = aSelection.Selection
interop.CimMdlrAPI.ICimEntity aEntity = (interop.CimMdlrAPI.ICimEntity)aEntList[1]
aIMdRound.Entity = aEntity
aMdRoundProcedure.Execute()
