IFilterAnd

Use this type of filter to construct a compound statement from two filters with an AND condition.

Namespace

interop.CimBaseAPI

Compatibility

Cimatron 2024 or later

Properties

None

Methods

void GetFilters ( out interop.CimBaseAPI.IEntityFilter,out interop.CimBaseAPI.IEntityFilter )
void SetFilters ( interop.CimBaseAPI.IEntityFilter, interop.CimBaseAPI.IEntityFilter )

Events

NoneÂ 

Remarks

None

Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
if (aCimApp != null)
{
interop.CimatronE.ICimDocument aDoc = (interop.CimatronE.ICimDocument)aCimApp.GetActiveDoc();
if (aDoc != null)
{
interop.CimMdlrAPI.IModelContainer ModelDoc = (interop.CimMdlrAPI.IModelContainer)aDoc;
interop.CimMdlrAPI.IModel TmpModel = (interop.CimMdlrAPI.IModel)ModelDoc.Model;
interop.CimMdlrAPI.IEntityQuery aModel = (interop.CimMdlrAPI.IEntityQuery)TmpModel;
interop.CimMdlrAPI.IEntityQuery QEntities = (interop.CimMdlrAPI.IEntityQuery)aModel;
interop.CimBaseAPI.FilterAnd filter = (interop.CimBaseAPI.FilterAnd)QEntities.CreateFilter(interop.CimMdlrAPI.EFilterEnumType.cmFilterAnd);


interop.CimBaseAPI.FilterAnd aIFilterAnd = (interop.CimBaseAPI.FilterAnd)filter;


interop.CimBaseAPI.FilterStyle aFilterStyle1 = (interop.CimBaseAPI.FilterStyle)QEntities.CreateFilter(interop.CimMdlrAPI.EFilterEnumType.cmFilterStyle);
aFilterStyle1.Add(0);

interop.CimBaseAPI.FilterColor aFilterStyle2 = (interop.CimBaseAPI.FilterColor)QEntities.CreateFilter(interop.CimMdlrAPI.EFilterEnumType.cmFilterColor);
aFilterStyle2.Add(80);

aIFilterAnd.SetFilters((interop.CimBaseAPI.IEntityFilter)aFilterStyle1, (interop.CimBaseAPI.IEntityFilter)aFilterStyle2);

interop.CimMdlrAPI.ISetsFactory SetFac = TmpModel.GetSetsFactory();
interop.CimMdlrAPI.ISet aSet = SetFac.CreateSet("iSetsName", (interop.CimMdlrAPI.IEntityFilter)aIFilterAnd);

interop.CimBaseAPI.IEntityFilter aFilter1;
interop.CimBaseAPI.IEntityFilter aFilter2;
aIFilterAnd.GetFilters(out aFilter1, out aFilter2);
}

aIFilterAnd = (interop.CimBaseAPI.FilterAnd)(filter)

aSet = (interop.CimMdlrAPI.ISet)(SetFac.CreateSet("iSetsName", (interop.CimMdlrAPI.IEntityFilter)(aIFilterAnd)))
