IMdSpline

This class represents a modeler procedure for creating Spline with the help of 3D point coordinates of points. It also provides a facility to make Spline closed. Spline can be created with two methods;

  1. Passing through points.
  2. With help of control points.

Example 1: Mode-1-Through PointsExample 1: Mode-1-Through Points // create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)CimApp.GetActiveDoc();
if (aDOC != null) // Check if document was found
{
// get model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
// Create the point procedure depended on active part document.
if (aDOC.Type == interop.CimBaseAPI.DocumentEnumType.cmPart)
{
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

interop.CimMdlrAPI.MdSpline aSplineProcedure =
(interop.CimMdlrAPI.MdSpline)
aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmSplineProcedure);

interop.CimMdlrAPI.IMdSpline aSplineProc = (interop.CimMdlrAPI.IMdSpline)aSplineProcedure;
aSplineProc.ThroughType = interop.CimMdlrAPI.SplineType.cmSplineThroughPoints;
aSplineProc.Mode = interop.CimMdlrAPI.SplineMode.cmSplineOpen;
double[] aPointsVec = new double[12];
aPointsVec[0] = 0;
aPointsVec[1] = 0;
aPointsVec[2] = 0;
aPointsVec[3] = 20;
aPointsVec[4] = 20;
aPointsVec[5] = 0;
aPointsVec[6] = 40;
aPointsVec[7] = 20;
aPointsVec[8] = 0;
aPointsVec[9] = 60;
aPointsVec[10] = 0;
aPointsVec[11] = 0;
aSplineProc.Points = aPointsVec;
aSplineProcedure.Execute();
}
}
