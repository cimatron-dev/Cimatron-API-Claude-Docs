IMdOffsetFace

Use this interface to create one or more faces that are offset from existing faces.

Namespace

interop.CimMdlrAPI

Compatibility

Cimatron 2026 or later

Properties

Get, Set

Entities

Example 1-Offset Face Global ValueExample 1-Offset Face Global Value
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;
// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;
//Create Offset Face procedure
interop.CimMdlrAPI.MdOffsetFace aMdOffsetFaceProcedure =
(interop.CimMdlrAPI.MdOffsetFace)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmOffsetFaceProcedure);
interop.CimMdlrAPI.IMdOffsetFace aIMdOffsetFace = (interop.CimMdlrAPI.IMdOffsetFace)aMdOffsetFaceProcedure;
interop.CimMdlrAPI.ICimEntity aFirstSelectedEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(25, aIModel);
interop.CimMdlrAPI.ICimEntity aSecondSelectedEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(28, aIModel);
interop.CimMdlrAPI.ICimEntity aThirdSelectedEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(36, aIModel);
interop.CimMdlrAPI.ICimEntityList aSelectedEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
aSelectedEntities.Add(aFirstSelectedEntity);
aSelectedEntities.Add(aSecondSelectedEntity);
aSelectedEntities.Add(aThirdSelectedEntity);
aIMdOffsetFace.Entities = aSelectedEntities;
aIMdOffsetFace.OffsetFaceExtendOption = interop.CimMdlrAPI.OffsetFaceExtendOption.cmExtendOption;
aIMdOffsetFace.OffsetFaceKeepRemoveOriginalOption = interop.CimMdlrAPI.OffsetFaceKeepRemoveOriginalOption.cmRemoveOriginalOption;
aIMdOffsetFace.GlobalOffsetValue = 3.5;
aIMdOffsetFace.OffsetFaceResultMode = interop.CimMdlrAPI.OffsetFaceResultMode.cmCompleteResultOption;
aIMdOffsetFace.OffsetFlipDirection = interop.CimMdlrAPI.OffsetFlipDirection.cmFlipInside;
aMdOffsetFaceProcedure.Execute();
}
}
}

import clr
#Adding references of required interop dlls
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")
clr.AddReference("interop.CimServicesAPI")
clr.AddReference("interop.CimBaseAPI")
clr.AddReference("interop.CimMdlrAPI")
#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI
#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()
#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())
aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)
#Create Offset Face procedure on Modeler Model
aMdOffsetFaceProcedure = (interop.CimMdlrAPI.MdOffsetFace)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmOffsetFaceProcedure))
aIMdOffsetFace = (interop.CimMdlrAPI.IMdOffsetFace)(aMdOffsetFaceProcedure)
aIModel = (interop.CimMdlrAPI.IModel)(aModelCont.Model)
#Get the faces with PId
aFirstSelectedEntity = aIModel.GetEntityById(25, aIModel)
aSecondSelectedEntity = aIModel.GetEntityById(28, aIModel)
aThirdSelectedEntity = aIModel.GetEntityById(36, aIModel)
aSelectedEntities = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
aSelectedEntities.Add(aFirstSelectedEntity)
aSelectedEntities.Add(aSecondSelectedEntity)
aSelectedEntities.Add(aThirdSelectedEntity)
#Set / assign required properties and methods
aIMdOffsetFace.Entities = aSelectedEntities
aIMdOffsetFace.OffsetFaceExtendOption = interop.CimMdlrAPI.OffsetFaceExtendOption.cmExtendOption
aIMdOffsetFace.OffsetFaceKeepRemoveOriginalOption = interop.CimMdlrAPI.OffsetFaceKeepRemoveOriginalOption.cmRemoveOriginalOption
aIMdOffsetFace.GlobalOffsetValue = 3.5
aIMdOffsetFace.OffsetFaceResultMode = interop.CimMdlrAPI.OffsetFaceResultMode.cmCompleteResultOption
aIMdOffsetFace.OffsetFlipDirection = interop.CimMdlrAPI.OffsetFlipDirection.cmFlipInside
aMdOffsetFaceProcedure.Execute()
