Example 1-SplitFacesByPointExample 1-SplitFacesByPoint
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;
interop.CimMdlrAPI.MdSplitFaces aMdSplitFacesProcedure =
(interop.CimMdlrAPI.MdSplitFaces)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmSplitFacesProcedure);
interop.CimMdlrAPI.IMdSplitFaces aIMdSplitFaces = (interop.CimMdlrAPI.IMdSplitFaces)aMdSplitFacesProcedure;
interop.CimMdlrAPI.ISelection aSelection = (interop.CimMdlrAPI.ISelection)aDoc;
interop.CimMdlrAPI.ICimEntityList aEntList = aSelection.Selection;
double[] aPos = { 0, 0, 0 };
aIMdSplitFaces.Entities = aEntList;
aIMdSplitFaces.SplittingPosition = aPos;
aIMdSplitFaces.PointSectionOption = interop.CimMdlrAPI.SplitFPointSection.cmSplitFCrossSection;

aMdSplitFacesProcedure.Execute();
}
}
}
Example 2-SplitFacesByEntityExample 2-SplitFacesByEntity
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;
// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;
interop.CimMdlrAPI.MdSplitFaces aMdSplitFacesProcedure =
(interop.CimMdlrAPI.MdSplitFaces)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmSplitFacesProcedure);
interop.CimMdlrAPI.IMdSplitFaces aIMdSplitFaces = (interop.CimMdlrAPI.IMdSplitFaces)aMdSplitFacesProcedure;
interop.CimMdlrAPI.ISelection aSelection = (interop.CimMdlrAPI.ISelection)aDoc;
interop.CimMdlrAPI.ICimEntityList aEntList = aSelection.Selection;
interop.CimMdlrAPI.ICimEntity aSplittingEntity = aIModel.GetEntityById(49, aIModel);
double[] aProjDir = { 0, -1, 0 };
aIMdSplitFaces.Entities = aEntList;
aIMdSplitFaces.SplittingEntity = aSplittingEntity;
// Check if the splitting entity is Edge then check for more parameters.
if (aSplittingEntity.Type == interop.CimMdlrAPI.EntityEnumType.cmEdge)
{
// If the splitting entity is curve
// cmSplitFProjectionNormal = 0
// cmSplitFProjectionDirection = 1
aIMdSplitFaces.ProjectionOption = interop.CimMdlrAPI.SplitFProjection.cmSplitFProjectionNormal;
if (aIMdSplitFaces.ProjectionOption == interop.CimMdlrAPI.SplitFProjection.cmSplitFProjectionDirection)
{
// This will work only if "iProjectionOption" is set to "suSplitFProjectionDirection"
if (aProjDir == null)
{
throw new SystemException("Missing parameters for split procedure.");
}
//If the splitting entity is curve
aIMdSplitFaces.ProjectionDir = aProjDir; // Set Vector for splitting
}
if (aIMdSplitFaces.ProjectionOption == interop.CimMdlrAPI.SplitFProjection.cmSplitFProjectionNormal)
{
// The Max Distance option will work only
// If the splitting entity is curve
// Only if SplitProjection = cmSplitProjectionNormal
aIMdSplitFaces.ProjMaxDistance = 800;
}
}
aMdSplitFacesProcedure.Execute();
}
}
}
