Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
interop.CimBaseAPI.ICimDocument aDoc = (interop.CimBaseAPI.ICimDocument)aCimApp.GetActiveDoc();
interop.CimBaseAPI.IInteractionSink aInteractionSink = (interop.CimBaseAPI.IInteractionSink)aDoc;
interop.CimBaseAPI.IInteraction aInteraction = aInteractionSink.CreateInteraction(interop.CimBaseAPI.InteractionType.cmToolServices);
interop.CimServicesAPI.ToolServices aToolServices = (interop.CimServicesAPI.ToolServices)aInteraction;
object aTool = aToolServices.Create(interop.CimServicesAPI.ToolType.cmContourBoundary);
interop.CimMdlrAPI.IContourBoundary aContourBoundaryTool = (interop.CimMdlrAPI.IContourBoundary)aTool;

import clr
#Adding references of required interop dlls
clr.AddReference("interop.CimAppAccess")
clr.AddReference("interop.CimatronE")
clr.AddReference("interop.CimServicesAPI")
clr.AddReference("interop.CimBaseAPI")
clr.AddReference("interop.CimMdlrAPI")

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = CimatronE.IApplication(aCimAppAccess.GetApplication())
aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aInteractionSink = (interop.CimBaseAPI.IInteractionSink)(aDoc)
aInteraction = aInteractionSink.CreateInteraction(interop.CimBaseAPI.InteractionType.cmToolServices)
aToolServices = (interop.CimServicesAPI.ToolServices)(aInteraction)
aTool = aToolServices.Create(interop.CimServicesAPI.ToolType.cmContourBoundary)
aContourBoundaryTool = (interop.CimMdlrAPI.IContourBoundary)(aTool)
