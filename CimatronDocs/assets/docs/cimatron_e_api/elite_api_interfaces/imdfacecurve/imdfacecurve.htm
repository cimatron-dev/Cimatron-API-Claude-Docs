IMdFaceCurve

This interface represents the Curves > From Face procedure inÂ Cimatron. Face curves can be created in 3 modes;

  • creating lines from face edges.
  • displaying curves and edges of the face.
  • by specifying point from which to take section or cross section.

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application

if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

interop.CimMdlrAPI.MdFaceCurve aMdFaceCurveProcedure =
(interop.CimMdlrAPI.MdFaceCurve)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmFaceCurveProcedure);

interop.CimMdlrAPI.IMdFaceCurve aIMdFaceCurve = (interop.CimMdlrAPI.IMdFaceCurve)aMdFaceCurveProcedure;
interop.CimMdlrAPI.ISelection aSelection = (interop.CimMdlrAPI.ISelection)aDoc;
interop.CimMdlrAPI.ICimEntityList aEntList = aSelection.Selection;
aIMdFaceCurve.Entity = (interop.CimMdlrAPI.ICimEntity)aEntList[1];
aIMdFaceCurve.Mode = interop.CimMdlrAPI.eFaceCurveMode.cmFaceCurveModeEdges;
aMdFaceCurveProcedure.Execute();
}
}
}

Example 3: Mode 3 -From pointExample 3: Mode 3 -From point
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

interop.CimMdlrAPI.MdFaceCurve aMdFaceCurveProcedure =
(interop.CimMdlrAPI.MdFaceCurve)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmFaceCurveProcedure);

interop.CimMdlrAPI.IMdFaceCurve aIMdFaceCurve = (interop.CimMdlrAPI.IMdFaceCurve)aMdFaceCurveProcedure;
interop.CimMdlrAPI.ICimEntity aFaceEnt = aIModel.GetEntityById(44, aIModel);
aIMdFaceCurve.Entity = aFaceEnt;
aIMdFaceCurve.Mode = interop.CimMdlrAPI.eFaceCurveMode.cmFaceCurveModePoint;

interop.CimMdlrAPI.ICimEntity aCurveEnt = aIModel.GetEntityById(41, aIModel);
interop.CimMdlrAPI.IGeometry3D aGeom = aCurveEnt.Geometry;
if (aGeom.Type == interop.CimMdlrAPI.GeomType.cmGeomCurve)
{
interop.CimServicesAPI.IGeom3DCurve aCurve = (interop.CimServicesAPI.IGeom3DCurve)aGeom;
//Get the midpoint IPointData to get the param value at midpoint.
interop.CimServicesAPI.IPointData aPointData =
(interop.CimServicesAPI.IPointData)aCurve.GetPointData(interop.CimServicesAPI.EPointType.cmPtMid);

double param = aPointData.FirstCurveParam;
//With the help of midpoint param we can get midpoint of curve
aIMdFaceCurve.Point = aCurve.Evaluate(param);

}
aIMdFaceCurve.SectionOption = 0;
aMdFaceCurveProcedure.Execute();
}
}
}
