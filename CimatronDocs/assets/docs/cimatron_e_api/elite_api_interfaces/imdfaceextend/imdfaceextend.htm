IMdFaceExtend

Use this interface to extend faces beyond their edges. This extends one or more faces (or creates new faces) on the edges of open or stitched faces.

Namespace

interop.CimMdlrAPI

Compatibility

Cimatron 2024 or later

Properties

Remarks

Prerequisites

  • Open the Cimatron Application and create a Solid Feature.
  • Consider the edge/curve to extend the Face

Example 1-Tangent NormalExample 1-Tangent Normal
// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
if (CimApp != null)
{
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Set model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the Sketcher procedure depended on the type of the active document.
// Get model type (can be Part/Assembly/NC)
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

//Create Face Extend type procedure
interop.CimMdlrAPI.MdFaceExtend aMdFaceExtendProcedure =
(interop.CimMdlrAPI.MdFaceExtend)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmFaceExtendProcedure);

interop.CimMdlrAPI.IMdFaceExtend aIMdFaceExtend = (interop.CimMdlrAPI.IMdFaceExtend)aMdFaceExtendProcedure;

interop.CimMdlrAPI.ICimEntity aFirstSelectedEntity = (interop.CimMdlrAPI.ICimEntity)aIModel.GetEntityById(13, aIModel);

interop.CimMdlrAPI.ICimEntityList aSelectedEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
//interop.CimMdlrAPI.ICimEntityList agetSelectedEntities = (interop.CimMdlrAPI.ICimEntityList)new interop.CimBaseAPI.CimEntityList();
aSelectedEntities.Add(aFirstSelectedEntity);

aIMdFaceExtend.Entities = aSelectedEntities;
aIMdFaceExtend.Option = interop.CimMdlrAPI.ExtendOptions.cmExtendTangent;
aIMdFaceExtend.ExtendNormalContinuousMode = interop.CimMdlrAPI.ExtendNormalContinuousMode.cmExtendNormal;
aIMdFaceExtend.Delta = 60;
aIMdFaceExtend.FlipExtensionDirection = 1;
aMdFaceExtendProcedure.Execute();
}
}

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)

#Create Extend Face procedure on Modeler Model
aMdFaceExtendProcedure = (interop.CimMdlrAPI.MdFaceExtend)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmFaceExtendProcedure))
aIMdFaceExtend = (interop.CimMdlrAPI.IMdFaceExtend)(aMdFaceExtendProcedure)

aIModel = (interop.CimMdlrAPI.IModel)(aModelCont.Model)

#Get the Edges with PId
aFirstSelectedEntity = aIModel.GetEntityById(13, aIModel)
aSecondSelectedEntity = aIModel.GetEntityById(15, aIModel)

aSelectedEntities = interop.CimMdlrAPI.ICimEntityList(interop.CimBaseAPI.CimEntityList())
aSelectedEntities.Add(aFirstSelectedEntity)
aSelectedEntities.Add(aSecondSelectedEntity)

#Set / assign required properties and methods
aIMdFaceExtend.Entities = aSelectedEntities
aIMdFaceExtend.Option = interop.CimMdlrAPI.ExtendOptions.cmExtendTangent
aIMdFaceExtend.ExtendNormalContinuousMode = interop.CimMdlrAPI.ExtendNormalContinuousMode.cmExtendNormal
aIMdFaceExtend.Delta = 60
aIMdFaceExtend.FlipExtensionDirection = 1
aMdFaceExtendProcedure.Execute()
