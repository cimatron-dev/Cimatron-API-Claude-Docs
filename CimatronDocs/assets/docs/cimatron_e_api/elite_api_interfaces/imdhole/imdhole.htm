interop.CimAppAccess.AppAccess aAppAcc = new interop.CimAppAccess.AppAccess();
interop.CimatronE.IApplication cimApp = null;
cimApp = aAppAcc.GetApplication();

interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)cimApp.GetActiveDoc();
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;

interop.CimMdlrAPI.IMdProcedure ProcedureHole =
(interop.CimMdlrAPI.MdHole)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmHoleProcedure);
interop.CimMdlrAPI.IMdHole IMdHole = (interop.CimMdlrAPI.IMdHole)ProcedureHole;

//Getting Created point
interop.CimMdlrAPI.IEntityQuery aQuery = (interop.CimMdlrAPI.IEntityQuery)aMdlrModel;
interop.CimMdlrAPI.IEntityFilter aEntityFilter = aQuery.CreateFilter(interop.CimMdlrAPI.EFilterEnumType.cmFilterEntityType);
interop.CimBaseAPI.FilterType aEntityTypeFilter = (interop.CimBaseAPI.FilterType)aEntityFilter;
aEntityTypeFilter.Add(interop.CimBaseAPI.EntityEnumType.cmPoint);
aQuery.SetFilter((interop.CimMdlrAPI.IEntityFilter)aEntityTypeFilter);
interop.CimMdlrAPI.ICimEntityList aPtList = aQuery.Select();
interop.CimMdlrAPI.ICimEntity aPt = aPtList[1];

IMdHole.Diameter = 10;
IMdHole.Depth = 90;
IMdHole.HoleOption = interop.CimMdlrAPI.HoleOption.cmHoleOneDir;
IMdHole.HeadType = interop.CimMdlrAPI.HoleHeadType.cmNoHead;
IMdHole.DrilledType = 0;
IMdHole.Entities = aPtList;

ProcedureHole.Execute();
