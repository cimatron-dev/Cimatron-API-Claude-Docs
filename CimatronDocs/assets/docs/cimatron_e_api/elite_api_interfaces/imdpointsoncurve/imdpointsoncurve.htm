IMdPointsOnCurve

This class represents a modeler procedure for creating points on curve at specific distances, or for dividing a curve into intervals with help of points created.

Example

Example 1:Mode 1 By Distance modeExample 1:Mode 1 By Distance mode // create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
if (CimApp != null)
{
// Get active document from application
interop.CimatronE.ICimDocument aDoc = CimApp.GetActiveDoc();
if (aDoc != null)
{
// Get model container from document
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDoc;

// Create the MdPointsOnCurve procedure depended on the type of the active document.
if (aDoc.Type == interop.CimatronE.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from document
interop.CimMdlrAPI.MdlrModel aMdlrModel =
(interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
interop.CimMdlrAPI.IModel aIModel = (interop.CimMdlrAPI.IModel)aModelCont.Model;

interop.CimMdlrAPI.MdPointsOnCurve aMdPointsOnCurveProcedure =
(interop.CimMdlrAPI.MdPointsOnCurve)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmPointsOnCurveProcedure);
//Get the curve entity with PId .
interop.CimMdlrAPI.ICimEntity aCurveEnt = aIModel.GetEntityById(18, aIModel);
interop.CimMdlrAPI.IMdPointsOnCurve aIMdPointsOnCurve = (interop.CimMdlrAPI.IMdPointsOnCurve)aMdPointsOnCurveProcedure;
//Set curve entity
aIMdPointsOnCurve.Entity = aCurveEnt;
aIMdPointsOnCurve.Mode = interop.CimMdlrAPI.MdPointsOnCurveMode.cmPointsOnCurveDistance;
aIMdPointsOnCurve.DirectionSide = 0;
aIMdPointsOnCurve.Distance = 5;
aIMdPointsOnCurve.Counter = 10;
interop.CimMdlrAPI.IGeometry3D aGeom = aCurveEnt.Geometry;
if (aGeom.Type == interop.CimMdlrAPI.GeomType.cmGeomCurve)
{
interop.CimServicesAPI.IGeom3DCurve aCurve = (interop.CimServicesAPI.IGeom3DCurve)aGeom;
//Get the midpoint IPointData to get the param value at midpoint.
interop.CimServicesAPI.IPointData aPointData =
(interop.CimServicesAPI.IPointData)aCurve.GetPointData(interop.CimServicesAPI.EPointType.cmPtMid);

double param = aPointData.FirstCurveParam;
//With the help of midpoint param we can get midpoint of curve
aIMdPointsOnCurve.ReferancePoint = aCurve.Evaluate(param);

}
aMdPointsOnCurveProcedure.Execute();
int aDirSide = aIMdPointsOnCurve.DirectionSide;
}
}
}

#importing modules from interop dlls
import interop.CimAppAccess
import interop.CimatronE
import interop.CimServicesAPI
import interop.CimBaseAPI
import interop.CimMdlrAPI

#Getting AppAccess Object
aCimAppAccess = interop.CimAppAccess.AppAccess()

#Getting instance of CimatronE Application
aCimApp = None
while (aCimApp == None) :
aCimApp = (interop.CimatronE.IApplication)(aCimAppAccess.GetApplication())

aDoc = (interop.CimBaseAPI.ICimDocument)(aCimApp.GetActiveDoc())
aModelCont = (interop.CimMdlrAPI.IModelContainer)(aDoc)
aMdlrModel = (interop.CimMdlrAPI.MdlrModel)(aModelCont.Model)


#Create MdPointsOnCurve procedure on Modeler Model
aMdPointsOnCurveProcedure = (interop.CimMdlrAPI.MdPointsOnCurve)(aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmPointsOnCurveProcedure))
aIMdPointsOnCurve = (interop.CimMdlrAPI.IMdPointsOnCurve)(aMdPointsOnCurveProcedure)

aIModel = (interop.CimMdlrAPI.IModel)(aModelCont.Model);
#Get the curve entity with PId .
aCurveEnt = (interop.CimMdlrAPI.ICimEntity)(aIModel.GetEntityById(20, aIModel))
#Set curve entity
aIMdPointsOnCurve.Entity = aCurveEnt
aIMdPointsOnCurve.Mode = interop.CimMdlrAPI.MdPointsOnCurveMode.cmPointsOnCurveDistance
aIMdPointsOnCurve.DirectionSide = 1
aIMdPointsOnCurve.Distance = 5
aIMdPointsOnCurve.Counter = 10
aGeom = (interop.CimMdlrAPI.IGeometry3D)(aCurveEnt.Geometry)
if aGeom.Type == interop.CimMdlrAPI.GeomType.cmGeomCurve :
aCurve = (interop.CimServicesAPI.IGeom3DCurve)(aGeom)
#Get the midpoint IPointData to get the param value at midpoint.
aPointData = (interop.CimServicesAPI.IPointData)(aCurve.GetPointData(interop.CimServicesAPI.EPointType.cmPtMid))
param = (aPointData.FirstCurveParam)
#With the help of midpoint param we can get midpoint of curve
aIMdPointsOnCurve.ReferancePoint = aCurve.Evaluate(param)

aMdPointsOnCurveProcedure.Execute()
