Remarks

Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication aCimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
interop.CimBaseAPI.IInteractionSink aInteractionSink = (interop.CimBaseAPI.IInteractionSink)aCimApp;
interop.CimBaseAPI.IInteraction aInteraction = aInteractionSink.CreateInteraction(interop.CimBaseAPI.InteractionType.cmUserControlDialog);
interop.CimBaseAPI.UserControlDialog aIUserControlDialog = (interop.CimBaseAPI.UserControlDialog)aInteraction;
aIUserControlDialog.Create();
aIUserControlDialog.Title = "Dialog_Title";
aIUserControlDialog.ControlXPos = 500;
aIUserControlDialog.ControlYPos = 200;
aIUserControlDialog.ControlHeight = 150;
aIUserControlDialog.ControlWidth = 200;
aIUserControlDialog.CoordinateMapping = interop.CimBaseAPI.ECoordinateMapping.cmCimatronWnd;
aIUserControlDialog.XPos = 500;
aIUserControlDialog.YPos = 100;
// Activate control
aIUserControlDialog.SetControl("E11_User_Control.UC11");
aIUserControlDialog.Titlebar = 1;
