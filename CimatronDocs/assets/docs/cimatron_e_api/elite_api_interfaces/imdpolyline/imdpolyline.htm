IMdPolyLine

This class represents a modeler procedure for creating a PolyLine using the 3D point coordinates of points. It also provides a facility to make PolyLine closed.

Example

// create Cimatron App Access object
interop.CimAppAccess.AppAccess aAppAccess = new interop.CimAppAccess.AppAccess();
//Get application object from App access
interop.CimatronE.IApplication CimApp = (interop.CimatronE.IApplication)aAppAccess.GetApplication();
// Get active document from application
interop.CimBaseAPI.ICimDocument aDOC = (interop.CimBaseAPI.ICimDocument)CimApp.GetActiveDoc();
if (aDOC != null) // Check if document was found
{
// get model container
interop.CimMdlrAPI.IModelContainer aModelCont = (interop.CimMdlrAPI.IModelContainer)aDOC;
// Create the point procedure depended on active part document.
if (aDOC.Type == interop.CimBaseAPI.DocumentEnumType.cmPart)
{
// If active document is part then get the MdlrModel from model container
interop.CimMdlrAPI.MdlrModel aMdlrModel = (interop.CimMdlrAPI.MdlrModel)aModelCont.Model;
//Create MdPolyLine procedure on Modeler Model
interop.CimMdlrAPI.MdPolyLine aPolyLineProcedure =
(interop.CimMdlrAPI.MdPolyLine)aMdlrModel.CreateProcedure(interop.CimMdlrAPI.MdProcedureType.cmPolyLineProcedure);

interop.CimMdlrAPI.IMdPolyLine aIMdPolyLine = (interop.CimMdlrAPI.IMdPolyLine)aPolyLineProcedure;
aIMdPolyLine.Closed=1;
double[] aPointVec = new double[9];
aPointVec[0] = 10;
aPointVec[1] = 0;
aPointVec[2] = 0;
aPointVec[3] = 100;
aPointVec[4] = 100;
aPointVec[5] = 0;
aPointVec[6] = 200;
aPointVec[7] = 0;
aPointVec[8] = 0;
aIMdPolyLine.Points = aPointVec;
aIMdPolyLine.Tolerance = 0.5;

aPolyLineProcedure.Execute();
}
}
